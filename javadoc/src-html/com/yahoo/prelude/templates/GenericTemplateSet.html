002package com.yahoo.prelude.templates;
003
004import com.yahoo.protect.Validator;
005import com.yahoo.search.Query;
006
007import java.io.Writer;
008import java.util.HashMap;
009import java.util.LinkedHashMap;
010import java.util.Properties;
011
012/**
013 * Superclass of a set of templates for rendering (serializing) results
014 * 
015 * @deprecated use a renderer instead
016 */
017@Deprecated // TODO: Remove on Vespa 7
018public class GenericTemplateSet {
019
020    public static final String  DEFAULT_MIMETYPE    = "text/xml";
021    public static final String  DEFAULT_ENCODING    = "utf-8";
022
023    /** Templates */
024    private HashMap<String, Template<? extends Writer>> templates;
025
026    /** The text MIME subtype this template returns, xml, plain or html */
027    private String mimeType;
028
029    /** The charset encoding this template should have */
030    private String encoding;
031
032    private String boldOpenTag = null;
033    private String boldCloseTag = null;
034    private String separatorTag = null;
035
036    /**
037     * Document summary class for this template
038     */
039    private String summaryClass = null;
040
041    /**
042     * The unique name of this template set
043     */
044    private final String name;
045
046    /**
047     * Creates a template set containing no templates
048     */
049    public GenericTemplateSet(String name, String mimeType, String encoding) {
050        this.mimeType    = mimeType;
051        this.encoding    = encoding;
052        this.name = name;
053
054        templates = new LinkedHashMap<>();
055    }
056
057
058    public String getName() {
059        return name;
060    }
061
062    /**
063     * Returns the text MIME
064     */
065    public String getMimeType() { return mimeType; }
066
067    /**
068     * Returns the text encoding
069     */
070    public String getEncoding() { return encoding; }
071
072    /** Returns the encoding of the query, or the encoding given by the template if none is set */
073    public final String getRequestedEncoding(Query query) {
074        String encoding = query.getModel().getEncoding();
075        if (encoding != null) return encoding;
076        return getEncoding();
077    }
078
079    /**
080     * Returns the selected template
081     *
082     * @return the template to use, never null
083     */
084    public Template<? extends Writer> getTemplate(String templateName) {
085      return templates.get(templateName);
086    }
087
088    /**
089     * Sets the selected template
090     *
091     * @throws NullPointerException if the given template is null
092     */
093    public void setTemplate(String templateName, Template<? extends Writer> template) {
094      templates.put(templateName,template);
095    }
096
097    /**
098     * Sets the selected template
099     *
100     * @throws NullPointerException if the given template is null
101     */
102    public void setTemplateNotNull(String templateName, Template<? extends Writer> template) {
103        Validator.ensureNotNull("Template "+templateName,template);
104        templates.put(templateName,template);
105    }
106
107
108    /**
109     * Sets the highligting marks for this template
110     *
111     * @param start the highlingting start mark
112     * @param end   the highlingting end mark
113     * @param sep   the highlingting separator mark
114     */
115    public void setHighlightTags(String start, String end, String sep) {
116        boldOpenTag = start;
117        boldCloseTag = end;
118        separatorTag = sep;
119    }
120
121    // may return null
122    public String getBoldOpenTag() {
123        return boldOpenTag;
124    }
125
126    // may return null
127    public String getBoldCloseTag() {
128        return boldCloseTag;
129    }
130
131    // may return null
132    public String getSeparatorTag() {
133        return separatorTag;
134    }
135
136
137    /**
138     * Set the default summary class to use with this template.
139     */
140    public void setSummaryClass(String summaryClass) {
141        this.summaryClass = summaryClass;
142    }
143
144    /**
145     * Type safe accessor to get the default document summary class for this
146     * template set.  This is also here to insulate the rest of the code
147     * against changes in the naming of the properties in the property file.
148     */
149    public String getSummaryClass() {
150        if (summaryClass != null && ! summaryClass.isEmpty()) {
151            return summaryClass;
152        } else {
153            return null;
154        }
155    }
156
157}
