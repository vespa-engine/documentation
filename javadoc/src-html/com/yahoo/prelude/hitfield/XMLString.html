004/**
005 * A representation of an XML chunk.
006 *
007 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
008 */
009public class XMLString {
010
011    private final String content;
012
013    public XMLString(String content) {
014        this.content = content;
015    }
016
017    public String toString() {
018        return content;
019    }
020
021}
