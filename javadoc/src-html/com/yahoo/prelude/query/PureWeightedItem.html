004import java.nio.ByteBuffer;

006/**
007 * @author baldersheim
008 */
009// TODO: Fix javadoc
010public abstract class PureWeightedItem extends Item {
011
012    public PureWeightedItem(int weight) {
013        setWeight(weight);
014    }
015    @Override
016    public void setIndexName(String index) {
017        // No index
018    }
019
020    @Override
021    public String getName() {
022        return getItemType().name();  //To change body of implemented methods use File | Settings | File Templates.
023    }
024
025    @Override
026    public int encode(ByteBuffer buffer) {
027        encodeThis(buffer);
028        return 1;
029    }
030
031    @Override
032    protected void appendBodyString(StringBuilder buffer) {
033        buffer.append(':').append(getWeight());
034    }
035}
