004/**
005 * A component version specification.
006 * <p>
007 * Version specifications have four components.
008 * <ol>
009 * <li>Major version. A non-negative integer.</li>
010 * <li>Minor version. A non-negative integer.</li>
011 * <li>Micro version. A non-negative integer.</li>
012 * <li>Qualifier. An ascii text string. See <code>Version(String)</code> for the
013 * format of the qualifier string.</li>
014 * </ol>
015 * <p>
016 * A null version component means "unspecified", i.e any value of that
017 * component matches the specification
018 * <p>
019 * <code>VersionSpecification</code> objects are immutable.
020 *
021 * @author arnej27959
022 * @author bratseth
023 */
024
025public final class VersionSpecification implements Comparable<VersionSpecification> {
026
027    private Integer major = null;
028    private Integer minor = null;
029    private Integer micro = null;
030    private String  qualifier = null;
031
032    private String stringValue;
033
034    /** The empty version */
035    public static final VersionSpecification emptyVersionSpecification = new VersionSpecification();
036
037    /** Creates an empty version */
038    public VersionSpecification() {
039        this(null, null, null, null);

069     * @param minor minor component of the version specification, or null if not specified
070     * @param micro micro component of the version specification, or null if not specified
071     * @throws IllegalArgumentException If the numerical components are
072     *         negative.
073     */
074    public VersionSpecification(Integer major, Integer minor, Integer micro) {
075        this(major, minor, micro, null);
076    }
077
078    /**
079     * Creates a version specification from the specifed components.
080     *
081     * @param major major component of the version specification, or null if not specified
082     * @param minor minor component of the version specification, or null if not specified
083     * @param micro micro component of the version specification, or null if not specified
084     * @param qualifier Qualifier component of the version specification, or null if not specified
085     * @throws IllegalArgumentException if the numerical components are negative
086     *         the qualifier string contains non-word/digit-characters, or
087     *         an earlier component is not specified but a later one is
088     */
089    public VersionSpecification(Integer major, Integer minor, Integer micro, String qualifier) {
090        this.major = major;
091        this.minor = minor;
092        this.micro = micro;
093        this.qualifier = qualifier;
094        initialize();
095    }
096
097    /**
098     * Creates a version specification from the specified string.
099     *
100     * <p>
101     * VersionSpecification strings follows this grammar (same as Osgi versions):
102     *
103     * <pre>
104     * version ::= major('.'minor('.'micro('.'qualifier)?)?)?
105     * major ::= digit+
106     * minor ::= digit+
107     * micro ::= digit+
108     * qualifier ::= (alpha|digit|'_'|'-')+
109     * digit ::= [0..9]
110     * alpha ::= [a..zA..Z]
111     * </pre>
112     *
113     * @param versionString String representation of the version specification
114     * @throws IllegalArgumentException If <code>version</code> is improperly
115     *         formatted.
116     */
117    public VersionSpecification(String versionString) {
118        if (! "".equals(versionString)) {
119            String[] components = versionString.split("\\x2e"); // Split on dot
120
121            if (components.length > 0) {
122                String s = components[0];
123                if (! s.equals("*")) {
124                    major = new Integer(s);
125                }
126            }
127            if (components.length > 1) {
128                String s = components[1];
129                if (! s.equals("*")) {
130                    minor = new Integer(s);
131                }
132            }
133            if (components.length > 2) {
134                String s = components[2];
135                if (! s.equals("*")) {
136                    micro = new Integer(s);
137                }
138            }
139            if (components.length > 3) {
140                qualifier = components[3];
141            }
142            if (components.length > 4)
143                throw new IllegalArgumentException("Too many components in " + versionString);
144        }
145        initialize();
146    }
147
148    public static VersionSpecification fromString(String versionString) {
149        if (versionString == null) {
150            return emptyVersionSpecification;
151        } else {
152            return new VersionSpecification(versionString);
153        }
154    }
155
156    /**
157     * Must be called on construction after the component values are set
158     *
159     * @throws IllegalArgumentException If the numerical components are negative
160     *         or the qualifier string is invalid.
161     */
162    private void initialize() {
163        stringValue = toStringValue(major, minor, micro, qualifier);
164        ensureUnspecifiedOnlyToTheRight(major, minor, micro, qualifier);
165
166        if (major!=null && major < 0)
167            throw new IllegalArgumentException("Negative major in " + this);
168        if (minor!=null && minor < 0)
169            throw new IllegalArgumentException("Negative minor in " + this);
170        if (micro!=null && micro < 0)
171            throw new IllegalArgumentException("Negative micro in " + this);
172
173        String q = getQualifier();
174        for (int i = 0; i < q.length(); i++) {
175            if ( !Character.isLetterOrDigit(q.charAt(i)) )
176                throw new IllegalArgumentException("Invalid qualifier in " + this +
177                                                   ": Invalid character at position " + i + " in qualifier");
178        }
179    }
180
181    private String toStringValue(Object... objects) {
182        StringBuilder b = new StringBuilder();
183        boolean first = true;
184        for (Object o : objects) {
185            if (o != null)  {
186                if (! first)
187                    b.append(".");
188                b.append(o);
189                first = false;
190            }
191        }
192        return b.toString();
193    }
194
195
196    private void ensureUnspecifiedOnlyToTheRight(Object... objects) {
197        boolean restMustBeNull=false;
198        for (Object o : objects) {
199            if (restMustBeNull && o!=null)
200                throw new IllegalArgumentException("A component to the left of a specified is unspecified in " + this);
201            if (o==null)
202                restMustBeNull=true;
203        }
204    }

238    /**
239     * Compares this <code>VersionSpecification</code> to another.
240     *
241     * <p>
242     * A version is considered to be <b>equal to </b> another version if the
243     * major, minor and micro components are equal and the qualifier component
244     * is equal (using <code>String.equals</code>).
245     * <p>
246     * Note that two versions are only equal if they are equally specified, use
247     * {@link #matches} to match a more specified version to a less specified.
248     *
249     * @param object The <code>VersionSpecification</code> object to be compared.
250     * @return <code>true</code> if <code>object</code> is a
251     *         <code>VersionSpecification</code> and is equal to this object;
252     *         <code>false</code> otherwise.
253     */
254    public boolean equals(Object object) {
255        if (object == this) return true;
256
257        if (!(object instanceof VersionSpecification)) return false;
258        VersionSpecification other = (VersionSpecification) object;
259        if ( ! equals(this.major,other.major)) return false;
260        if ( ! equals(this.minor,other.minor)) return false;
261        if ( ! equals(this.micro,other.micro)) return false;
262        if ( ! equals(this.qualifier,other.qualifier)) return false;
263        return true;
264    }
265
266    private boolean equals(Object o1,Object o2) {
267        if (o1==null && o2==null) return true;
268        if (o1==null || o2==null) return false;
269        return o1.equals(o2);
270    }

272    /**
273     * check if the Version specification is equal to the
274     * empty spec ("match anything")
275     **/
276    public boolean isEmpty() {
277        return equals(emptyVersionSpecification);
278    }

280    /**
281     * Returns the lowest possible Version object that matches this spec
282     **/
283    public Version lowestMatchingVersion() {
284        return new Version(getMajor(), getMinor(), getMicro(), getQualifier());
285    }

287    /**
288     * Returns true if the given version matches this specification.
289     * It matches if all the numeric components specified are the same
290     * as in the version, and both qualifiers are either null or set
291     * to the same value.  I.e, a version which includes a qualifier
292     * will only match exactly and will never return true from a
293     * request for an unspecified qualifier.
294     */
295    public boolean matches(Version version) {
296        if (matches(this.major, version.getMajor()) &&
297            matches(this.minor, version.getMinor()) &&
298            matches(this.micro, version.getMicro()))
299        {
300            return (version.getQualifier().equals(this.getQualifier()));
301        } else {
302            return false;
303        }
304    }
305
306    private boolean matches(Integer componentSpec, Integer component) {
307        if (componentSpec == null) return true;
308        return componentSpec.equals(component);
309    }
310
311    /**
312     * Compares this <code>VersionSpecification</code> object to another.
313     * <p>
314     * A version is considered to be <b>less than </b> another version if its
315     * major component is less than the other version's major component, or the
316     * major components are equal and its minor component is less than the other
317     * version's minor component, or the major and minor components are equal
318     * and its micro component is less than the other version's micro component,
319     * or the major, minor and micro components are equal and it's qualifier
320     * component is less than the other version's qualifier component (using
321     * <code>String.compareTo</code>).
322     * <p>
323     * A version is considered to be <b>equal to</b> another version if the
324     * major, minor and micro components are equal and the qualifier component
325     * is equal (using <code>String.compareTo</code>).
326     * <p>
327     * Unspecified numeric components are treated as 0, unspecified qualifier is treated as the empty string.
328     *
329     * @param other the <code>VersionSpecification</code> object to be compared.
330     * @return A negative integer, zero, or a positive integer if this object is
331     *         less than, equal to, or greater than the specified <code>VersionSpecification</code> object.
332     * @throws ClassCastException if the specified object is not a <code>VersionSpecification</code>.
333     */
334    public int compareTo(VersionSpecification other) {
335        int result = this.getMajor() - other.getMajor();
336        if (result != 0) return result;
337
338        result = this.getMinor() - other.getMinor();
339        if (result != 0) return result;
340
341        result = this.getMicro() - other.getMicro();
342        if (result != 0) return result;
343
344        return getQualifier().compareTo(other.getQualifier());
345    }
346
347    public VersionSpecification intersect(VersionSpecification other) {
348        return new VersionSpecification(
349                intersect(major, other.major, "major"),
350                intersect(minor, other.minor, "minor"),
351                intersect(micro, other.micro, "micro"),
352                intersect(qualifier, other.qualifier, "qualifier"));
353    }
354
355    private <T> T intersect(T spec1, T spec2, String name) {
356        if (spec1 == null) {
357            return spec2;
358        } else if (spec2 == null) {
359            return spec1;
360        } else if (spec1.equals(spec2)) {
361            return spec1;
362        } else {
363            throw new RuntimeException("The " + name + " component does not match(" + spec1 + "!=" + spec2 + ").");
364        }
365    }
