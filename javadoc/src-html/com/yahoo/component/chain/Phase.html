002package com.yahoo.component.chain;
003
004import com.yahoo.component.chain.dependencies.Dependencies;
005import net.jcip.annotations.Immutable;
006
007import java.util.Set;
008import java.util.TreeSet;
009
010/**
011 * Used for many to many constraints on searcher ordering.
012 *
013 * @author tonytv
014 */
015@Immutable
016public class Phase {
017    public final Dependencies dependencies;
018
019    public Phase(String name, Set<String> before, Set<String> after) {
020        dependencies = new Dependencies(provides(name), before, after);
021    }
022
023    public Phase(String name, Dependencies dependencies) {
024        this(name, dependencies.before(), dependencies.after());
025        assert(dependencies.provides().isEmpty());
026    }
027
028    private Set<String> provides(String name) {
029        Set<String> provides = new TreeSet<>();
030        provides.add(name);
031        return provides;
032    }
033
034    public String getName() {
035        return dependencies.provides().iterator().next();
036    }
037
038    public Set<String> before() {
039        return dependencies.before();
040    }
041
042    public Set<String> after() {
043        return dependencies.after();
044    }
045
046    public Phase union(Phase phase) {
047        assert(getName().equals(phase.getName()));
048
049        Dependencies union = dependencies.union(phase.dependencies);
050        return new Phase(getName(), union.before(), union.after());
051    }
052}
