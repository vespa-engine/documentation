002package com.yahoo.searchlib.rankingexpression.evaluation;
003
004import com.yahoo.searchlib.rankingexpression.rule.Function;
005import com.yahoo.searchlib.rankingexpression.rule.TruthOperator;
006
007/**
008 * A double value result of a ranking expression evaluation.
009 * In a boolean context doubles are true if they are different from 0.0
010 *
011 * @author bratseth
012 * @since  5.1.5
013 */
014public final class DoubleValue extends DoubleCompatibleValue {
015
016    // A note on performance: Reusing double values like below is actually slightly slower per evaluation,
017    // but the reduced garbage cost seems to regain this plus some additional percentages
018
019    private double value;
020
021    /** The double value instance for 0 */
022    public final static DoubleValue zero = DoubleValue.frozen(0);
023
024    public DoubleValue(double value) {
025        this.value = value;
026    }
027
028    /**
029     * Create a double which is frozen at the outset.
030     */
031    public static DoubleValue frozen(double value) {
032        DoubleValue doubleValue = new DoubleValue(value);
033        doubleValue.freeze();
034        return doubleValue;
035    }
036
037    @Override
038    public double asDouble() { return value; }
039
040    @Override
041    public DoubleValue asDoubleValue() { return this; }
042
043    @Override
044    public boolean asBoolean() { return value != 0.0; }
045
046    @Override
047    public DoubleValue negate() {
048        return mutable(-value);
049    }
050
051    @Override
052    public Value add(Value value) {
053        if (value instanceof TensorValue)
054            return value.add(this);
055
056        try {
057            return mutable(this.value + value.asDouble());
058        }
059        catch (UnsupportedOperationException e) {
060            throw unsupported("add",value);
061        }
062    }
063
064    @Override
065    public Value subtract(Value value) {
066        if (value instanceof TensorValue)
067            return value.negate().add(this);
068
069        try {
070            return mutable(this.value - value.asDouble());
071        }
072        catch (UnsupportedOperationException e) {
073            throw unsupported("subtract",value);
074        }
075    }
076
077    @Override
078    public Value multiply(Value value) {
079        if (value instanceof TensorValue)
080            return value.multiply(this);
081
082        try {
083            return mutable(this.value * value.asDouble());
084        }
085        catch (UnsupportedOperationException e) {
086            throw unsupported("multiply", value);
087        }
088    }
089
090    @Override
091    public Value divide(Value value) {
092        try {
093            return mutable(this.value / value.asDouble());
094        }
095        catch (UnsupportedOperationException e) {
096            throw unsupported("divide",value);
097        }
098    }
099
100    @Override
101    public Value function(Function function, Value value) {
102        // use the tensor implementation of max and min if the argument is a tensor
103        if ( (function.equals(Function.min) || function.equals(Function.max)) && value instanceof TensorValue)
104            return value.function(function, this);
105
106        try {
107            return mutable(function.evaluate(this.value, value.asDouble()));
108        }
109        catch (UnsupportedOperationException e) {
110            throw unsupported("function " + function.toString(), value);
111        }
112    }
113
114    private UnsupportedOperationException unsupported(String operation, Value value) {
115        return new UnsupportedOperationException("Cannot perform " + operation + " on " + value + " and " + this);
116    }
117
118    /** Returns this or a mutable copy assigned the given value */
119    private DoubleValue mutable(double value) {
120        DoubleValue mutable=this.asMutable();
121        mutable.value=value;
122        return mutable;
123    }
124
125    @Override
126    public DoubleValue asMutable() {
127        if ( ! isFrozen()) return this;
128        return new DoubleValue(value);
129    }
130
131    @Override
132    public String toString() {
133        return String.valueOf(value);
134    }
135
136    @Override
137    public boolean equals(Object other) {
138        if (this==other) return true;
139        if ( ! (other instanceof DoubleValue)) return false;
140        return ((DoubleValue)other).value==this.value;
141    }
142
143    @Override
144    public int hashCode() {
145        return toString().hashCode();
146    }
147
148}
