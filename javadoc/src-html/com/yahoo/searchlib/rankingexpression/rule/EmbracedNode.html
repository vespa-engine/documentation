002package com.yahoo.searchlib.rankingexpression.rule;
003
004import com.yahoo.searchlib.rankingexpression.evaluation.Context;
005import com.yahoo.searchlib.rankingexpression.evaluation.Value;
006
007import java.util.Collections;
008import java.util.Deque;
009import java.util.List;
010
011/**
012 * This class represents another expression enclosed in braces.
013 *
014 * @author Simon Thoresen
015 */
016public final class EmbracedNode extends CompositeNode {
017
018    // The node to embrace.
019    private final ExpressionNode value;
020
021    /**
022     * Creates a new expression node that embraces another.
023     *
024     * @param value The node to embrace.
025     */
026    public EmbracedNode(ExpressionNode value) {
027        this.value=value;
028    }
029
030    /** Returns the node enclosed by this */
031    public ExpressionNode getValue() { return value; }
032
033    @Override
034    public List<ExpressionNode> children() {
035        return Collections.singletonList(value);
036    }
037
038    @Override
039    public String toString(SerializationContext context, Deque<String> path, CompositeNode parent) {
040        String expression = value.toString(context, path, this);
041        if (value instanceof ReferenceNode)  return expression;
042        return  "(" + expression + ")";
043    }
044
045    @Override
046    public Value evaluate(Context context) {
047        return value.evaluate(context);
048    }
049
050    @Override
051    public CompositeNode setChildren(List<ExpressionNode> newChildren) {
052        if (newChildren.size() != 1)
053            throw new IllegalArgumentException("Expected 1 child but got " + newChildren.size());
054        return new EmbracedNode(newChildren.get(0));
055    }
056
057}
