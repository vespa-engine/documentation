002package com.yahoo.protect;
003
004
005import java.util.Map;
006import java.util.logging.Level;
007import java.util.logging.Logger;
008
009
010/**
011 * A class for interacting with the global state of the running VM.
012 *
013 * @author Steinar Knutsen
014 */
015public final class Process {
016
017    private static final Logger log = Logger.getLogger(Process.class.getName());
018
019    /** Die with a message, without dumping thread state */
020    public static void logAndDie(String message) {
021        logAndDie(message, null);
022    }
023
024    /** Die with a message, optionally dumping thread state */
025    public static void logAndDie(String message, boolean dumpThreads) {
026        logAndDie(message, null, dumpThreads);
027    }
028
029    /** Die with a message containing an exception, without dumping thread state */
030    public static void logAndDie(String message, Throwable thrown) {
031        logAndDie(message, thrown, false);
032    }
033
034    /**
035     * Log message as severe error, then forcibly exit runtime, without running
036     * exit handlers or otherwise waiting for cleanup.
037     *
038     * @param message message to log before exit
039     * @param thrown the throwable that caused the application to exit.
040     * @param dumpThreads if true the stack trace of all threads is dumped to the
041     *                   log with level info before shutting down
042     */
043    public static void logAndDie(String message, Throwable thrown, boolean dumpThreads) {
044        try {
045            if (dumpThreads)
046                dumpThreads();
047            if (thrown != null)
048                log.log(Level.SEVERE, message, thrown);
049            else
050                log.log(Level.SEVERE, message);
051        } finally {
052            try {
053                Runtime.getRuntime().halt(1);
054            }
055            catch (Throwable t) {
056                log.log(Level.SEVERE, "Runtime.halt rejected. Throwing an error.");
057                throw new ShutdownError("Shutdown requested, but failed to shut down");
058            }
059        }
060    }
061
062
063    private static void dumpThreads() {
064        try {
065            log.log(Level.INFO, "About to shut down. Commencing full thread dump for diagnosis.");
066            Map<Thread, StackTraceElement[]> allStackTraces = Thread.getAllStackTraces();
067            for (Map.Entry<Thread, StackTraceElement[]> e : allStackTraces.entrySet()) {
068                Thread t = e.getKey();
069                StackTraceElement[] stack = e.getValue();
070                StringBuilder forOneThread = new StringBuilder();
071                int initLen;
072                forOneThread.append("Stack for thread: ").append(t.getName()).append(": ");
073                initLen = forOneThread.length();
074                for (StackTraceElement s : stack) {
075                    if (forOneThread.length() > initLen) {
076                        forOneThread.append(" ");
077                    }
078                    forOneThread.append(s.toString());
079                }
080                log.log(Level.INFO, forOneThread.toString());
081            }
082            log.log(Level.INFO, "End of diagnostic thread dump.");
083        } catch (Exception e) {
084            // just give up...
085        }
086    }
087
088    @SuppressWarnings("serial")
089    public static class ShutdownError extends Error {
090
091        public ShutdownError(String message) {
092            super(message);
093        }
094
095    }
096
097}
