037     * Returns the first instance of the given class in this chain, or null if none
038     */
039    @SuppressWarnings("unchecked")
040    public final <T extends Properties> T getInstance(Class<T> propertyClass) {
041        if (propertyClass.isAssignableFrom(this.getClass())) return (T) this;
042        if (chained == null) return null;
043        return chained.getInstance(propertyClass);
044    }

088    /**
089     * Returns a snapshot of all properties of this having a given path prefix
090     * <p>
091     * Some sources of properties may not be list-able (e.g those using reflection)
092     * and will not be included in this snapshot.
093     *
094     *
095     * @param path         the prefix (up to a ".") of the properties to return, or null or the empty string to return all properties
096     * @param context      the context used to resolve the properties, or null if none
097     * @param substitution the properties which will be used to do string substitution in the values added to the map
098     */
099    public Map<String, Object> listProperties(CompoundName path, Map<String, String> context, Properties substitution) {
100        if (path == null)
101            path = CompoundName.empty;
102        if (chained() == null)
103            return new HashMap<>();
104        else
105            return chained().listProperties(path, context, this);
106    }

295    /**
296     * Converts a value to boolean - this will be true only if the value is either the empty string,
297     * or any Object which has a toString which is case-insensitive equal to "true"
298     */
299    protected final boolean asBoolean(Object value, boolean defaultValue) {
300        if (value == null) return defaultValue;
301
302        String s = value.toString();
303        int sz = s.length();
304        switch (sz) {
305            case 0:
306                return true;
307            case 4:
308                return ((s.charAt(0) | 0x20) == 't') &&
309                        ((s.charAt(1) | 0x20) == 'r') &&
310                        ((s.charAt(2) | 0x20) == 'u') &&
311                        ((s.charAt(3) | 0x20) == 'e');
312        }
313        return false;
314    }

409    protected final Integer asInteger(Object value, Integer defaultValue) {
410        try {
411            if (value == null)
412                return defaultValue;
413
414            if (value instanceof Integer)
415                return (Integer) value;
416
417            String stringValue = value.toString();
418            if (stringValue.isEmpty())
419                return defaultValue;
420
421            return new Integer(stringValue);
422        } catch (IllegalArgumentException e) {
423            throw new NumberFormatException("Not a valid integer");
424        }

559    /**
560     * Clones this instance and recursively all chained instance.
561     * Implementations should call this and clone their own state as appropriate
562     */
563    public Properties clone() {
564        try {
565            Properties clone = (Properties) super.clone();
566            if (chained != null)
567                clone.chained = this.chained.clone();
568            return clone;
569        } catch (CloneNotSupportedException e) {
570            throw new RuntimeException("Will never happen");
571        }
572    }

574    /**
575     * Clones a map by deep cloning each value which is cloneable and shallow copying all other values.
576     */
577    public static Map<CompoundName, Object> cloneMap(Map<CompoundName, Object> map) {
578        return cloneHelper.cloneMap(map);
579    }
580    /** Clones this object if it is clonable, and the clone is public. Returns null if not */
581    public static Object clone(Object object) {
582        return cloneHelper.clone(object);
