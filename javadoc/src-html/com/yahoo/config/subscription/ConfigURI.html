002package com.yahoo.config.subscription;
003
004import java.io.File;
005
006import com.yahoo.config.subscription.impl.JRTConfigRequester;
007
008/**
009 * A Config URI is a class that can be used to encapsulate a config source and a config id into one
010 * object to simplify parameter passing.
011 *
012 * @author lulf
013 * @since 5.1
014 */
015public class ConfigURI {
016    private String configId;
017    private ConfigSource source;
018
019    private ConfigURI(String configId, ConfigSource source) {
020        this.configId = configId;
021        this.source = source;
022    }
023
024    public String getConfigId() {
025        return configId;
026    }
027
028    public ConfigSource getSource() {
029        return source;
030    }
031
032    public static ConfigURI createFromId(String configId) {
033        return new ConfigURI(getConfigId(configId), getConfigSource(configId));
034    }
035
036    private static ConfigSource getConfigSource(String configId) {
037        if (configId.startsWith("file:")) {
038            return new FileSource(new File(configId.substring(5)));
039        } else if (configId.startsWith("dir:")) {
040            return new DirSource(new File(configId.substring(4)));
041        } else {
042            return JRTConfigRequester.defaultSourceSet;
043        }
044    }
045
046    private static String getConfigId(String configId) {
047        if (configId.startsWith("file:") || configId.startsWith("dir:")) {
048            return "";
049        } else {
050            return configId;
051        }
052    }
053
054    public static ConfigURI createFromIdAndSource(String configId, ConfigSource source) {
055        return new ConfigURI(configId, source);
056    }
057}
