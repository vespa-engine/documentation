002package com.yahoo.config.subscription;
003
004import java.util.jar.JarFile;
005
006/**
007 * Source specifying config as a jar file entry
008 * @author vegardh
009 * @since 5.1
010 *
011 */
012public class JarSource implements ConfigSource {
013    private final String path;
014    private final JarFile jarFile;
015
016    /**
017     * Creates a new jar source
018     * @param jarFile the jar file to use as a source
019     * @param path the path within the jar file, or null to use the default config/
020     */
021    public JarSource(JarFile jarFile, String path) {
022        this.path = path;
023        this.jarFile = jarFile;
024    }
025
026    public JarFile getJarFile() {
027        return jarFile;
028    }
029
030    public String getPath() {
031        return path;
032    }
033
034}
