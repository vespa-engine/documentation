002package com.yahoo.config.provision;
003
004import com.yahoo.slime.ArrayTraverser;
005import com.yahoo.slime.Cursor;
006import com.yahoo.slime.Inspector;
007import com.yahoo.slime.Slime;
008import com.yahoo.vespa.config.SlimeUtils;
009
010import java.io.IOException;
011import java.util.*;
012import java.util.logging.Logger;
013
014/**
015 * Information about provisioned hosts, and (de)serialization (from)to JSON.
016 *
017 * @author lulf
018 * @since 5.12
019 */
020public class ProvisionInfo {
021
022    private static final String mappingKey = "mapping";
023    private static final String hostSpecKey = "hostSpec";
024    private static final String hostSpecHostName = "hostName";
025    private static final String hostSpecMembership = "membership";
026    private static final String dockerImage = "dockerImage";
027
028    private final Set<HostSpec> hosts = new LinkedHashSet<>();
029
030    private ProvisionInfo(Set<HostSpec> hosts) {
031        this.hosts.addAll(hosts);
032    }
033
034    public static ProvisionInfo withHosts(Set<HostSpec> hosts) {
035        return new ProvisionInfo(hosts);
036    }
037
038    private void toSlime(Cursor cursor) {
039        Cursor array = cursor.setArray(mappingKey);
040        for (HostSpec host : hosts) {
041            Cursor object = array.addObject();
042            serializeHostSpec(object.setObject(hostSpecKey), host);
043        }
044    }
045
046    private void serializeHostSpec(Cursor cursor, HostSpec host) {
047        cursor.setString(hostSpecHostName, host.hostname());
048        if (host.membership().isPresent()) {
049            cursor.setString(hostSpecMembership, host.membership().get().stringValue());
050            if (host.membership().get().cluster().dockerImage().isPresent())
051                cursor.setString(dockerImage, host.membership().get().cluster().dockerImage().get());
052        }
053    }
054
055    public Set<HostSpec> getHosts() {
056        return Collections.unmodifiableSet(hosts);
057    }
058
059    private static ProvisionInfo fromSlime(Inspector inspector) {
060        Inspector array = inspector.field(mappingKey);
061        final Set<HostSpec> hosts = new LinkedHashSet<>();
062        array.traverse(new ArrayTraverser() {
063            @Override
064            public void entry(int i, Inspector inspector) {
065                hosts.add(createHostSpec(inspector.field(hostSpecKey)));
066            }
067        });
068        return new ProvisionInfo(hosts);
069    }
070
071    private static HostSpec createHostSpec(Inspector object) {
072        Optional<ClusterMembership> membership =
073            object.field(hostSpecMembership).valid() ? Optional.of(readMembership(object)) : Optional.empty();
074        HostSpec h = new HostSpec(object.field(hostSpecHostName).asString(), Collections.emptyList(), Optional.empty(), membership);
075        return h;
076    }
077
078    private static ClusterMembership readMembership(Inspector object) {
079        return ClusterMembership.from(object.field(hostSpecMembership).asString(),
080                                      object.field(dockerImage).valid() ? Optional.of(object.field(dockerImage).asString()) : Optional.empty());
081    }
082
083    public byte[] toJson() throws IOException {
084        Slime slime = new Slime();
085        toSlime(slime.setObject());
086        return SlimeUtils.toJsonBytes(slime);
087    }
088
089    public static ProvisionInfo fromJson(byte[] json) {
090        return fromSlime(SlimeUtils.jsonToSlime(json).get());
091    }
092
093    public ProvisionInfo merge(ProvisionInfo provisionInfo) {
094        Set<HostSpec> mergedSet = new LinkedHashSet<>();
095        mergedSet.addAll(this.hosts);
096        mergedSet.addAll(provisionInfo.getHosts());
097        return ProvisionInfo.withHosts(mergedSet);
098    }
099
100}
