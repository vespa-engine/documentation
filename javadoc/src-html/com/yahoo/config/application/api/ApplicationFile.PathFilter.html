002package com.yahoo.config.application.api;
003
004import com.yahoo.path.Path;
005
006import java.io.*;
007import java.util.ArrayList;
008import java.util.List;
009
010/**
011 * An application file represents a file within an application package. This class can be used to traverse the entire
012 * application package file structure, as well as read and write files to it, and create directories.
013 *
014 * @author Ulf Lillengen
015 */
016public abstract class ApplicationFile implements Comparable<ApplicationFile> {
017
018    private static final String metaDir = ".meta";
019    public static final String ContentStatusNew = "new";
020    public static final String ContentStatusChanged = "changed";
021    public static final String ContentStatusDeleted = "deleted";
022    protected final Path path;
023    private static final PathFilter defaultFilter = path1 -> true;
024
025    protected ApplicationFile(Path path) {
026        this.path = path;
027    }
028
029    /**
030     * Check whether or not this file is a directory.
031     *
032     * @return true if it is, false if not.
033     */
034    public abstract boolean isDirectory();
035
036    /**
037     * Test whether or not this file exists.
038     *
039     * @return true if it exists, false if not.
040     */
041    public abstract boolean exists();
042
043    /**
044     * Create a {@link Reader} for the contents of this file.
045     *
046     * @return A {@link Reader} that should be closed after use.
047     * @throws FileNotFoundException if the file is not found.
048     */
049    public abstract Reader createReader() throws FileNotFoundException;
050
051
052    /**
053     * Create an {@link InputStream} for the contents of this file.
054     *
055     * @return An {@link InputStream} that should be closed after use.
056     * @throws FileNotFoundException if the file is not found.
057     */
058    public abstract InputStream createInputStream() throws FileNotFoundException;
059
060    /**
061     * Create a directory at the path represented by this file. Parent directories will
062     * be automatically created.
063     *
064     * @return this
065     * @throws IllegalArgumentException if the directory already exists.
066     */
067    public abstract ApplicationFile createDirectory();
068
069    /**
070     * Write the contents from this reader to this file. Any existing content will be overwritten!
071     *
072     * @param input A reader pointing to the content that should be written.
073     * @return this
074     */
075    public abstract ApplicationFile writeFile(Reader input);
076
077    /**
078     * List the files under this directory. If this is file, an empty list is returned.
079     * Only immediate files/subdirectories are returned.
080     *
081     * @return a list of files in this directory.
082     */
083    public List<ApplicationFile> listFiles() {
084        return listFiles(defaultFilter);
085    }
086
087    /**
088     * List the files under this directory. If this is file, an empty list is returned.
089     * Only immediate files/subdirectories are returned.
090     *
091     * @param filter A filter functor for filtering path names
092     * @return a list of files in this directory.
093     */
094    public abstract List<ApplicationFile> listFiles(PathFilter filter);
095
096    /**
097     * List the files in this directory, optionally list files for subdirectories recursively as well.
098     *
099     * @param recurse Set to true if all files in the directory tree should be returned.
100     * @return a list of files in this directory.
101     */
102    public List<ApplicationFile> listFiles(boolean recurse) {
103        List<ApplicationFile> ret = new ArrayList<>();
104        List<ApplicationFile> files = listFiles();
105        ret.addAll(files);
106        if (recurse) {
107            for (ApplicationFile file : files) {
108                if (file.isDirectory()) {
109                    ret.addAll(file.listFiles(recurse));
110                }
111            }
112        }
113        return ret;
114    }
115
116    /**
117     * Delete the file pointed to by this. If it is a non-empty directory, the operation will throw.
118     *
119     * @return this.
120     * @throws RuntimeException if the file is a directory and not empty.
121     */
122    public abstract ApplicationFile delete();
123
124    /**
125     * Get the path that this file represents.
126     *
127     * @return a Path
128     */
129    public Path getPath() {
130        return path;
131    }
132
133    @Override
134    public String toString() {
135        return path.toString();
136    }
137
138    @Override
139    public boolean equals(Object other) {
140        if (other instanceof ApplicationFile) {
141            return path.equals(((ApplicationFile) other).path);
142        }
143        return false;
144    }
145
146    protected Path getMetaPath() {
147        if (path.toString().equals("")) {
148            return Path.fromString(metaDir).append(".root");
149        } else {
150            return path.getParentPath().append(metaDir).append(path.getName());
151        }
152    }
153
154    public abstract MetaData getMetaData();
155
156    public static class MetaData {
157
158        public String status = "unknown";
159        public String md5 = "";
160
161        public MetaData() {  }
162
163        public MetaData(String status, String md5) {
164            this.status = status;
165            this.md5= md5;
166        }
167
168        public String getStatus() {
169            return status;
170        }
171
172        public String getMd5() {
173            return md5;
174        }
175    }
176
177    public interface PathFilter {
178        boolean accept(Path path);
179    }
180
181}
