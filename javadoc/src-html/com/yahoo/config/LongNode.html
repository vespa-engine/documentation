002package com.yahoo.config;
003
004import edu.umd.cs.findbugs.annotations.NonNull;
005
006/**
007 * Represents a long in a configuration tree.
008 * @author gjoranv
009 */
010public class LongNode extends LeafNode<Long> {
011
012    public LongNode() {
013    }
014
015    public LongNode(long value) {
016        super(true);
017        this.value = value;
018    }
019
020    public Long value() {
021        return value;
022    }
023
024    @Override
025    public String getValue() {
026        return "" + value;
027    }
028
029    @Override
030    public String toString() {
031        return getValue();
032    }
033
034    @Override
035    protected boolean doSetValue(@NonNull String value) {
036        try {
037            this.value = Long.parseLong(value);
038            return true;
039        } catch (NumberFormatException e) {
040            return false;
041        }
042    }
043
044    @Override
045    void serialize(String name, Serializer serializer) {
046        serializer.serialize(name, value);
047    }
048
049    @Override
050    void serialize(Serializer serializer) {
051        serializer.serialize(value);
052    }
053}
