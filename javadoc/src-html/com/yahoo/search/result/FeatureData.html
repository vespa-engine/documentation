002package com.yahoo.search.result;
003
004import com.yahoo.data.access.Inspector;
005import com.yahoo.data.access.Inspectable;
006import com.yahoo.data.access.Type;
007import com.yahoo.data.JsonProducer;
008import com.yahoo.data.access.simple.JsonRender;
009
010/**
011 * A wrapper for structured data representing feature values.
012 */
013public class FeatureData implements Inspectable, JsonProducer {
014
015    private final Inspector value;
016
017    public FeatureData(Inspector value) {
018        this.value = value;
019    }
020
021    @Override
022    public Inspector inspect() {
023        return value;
024    }
025
026    public String toString() {
027        if (value.type() == Type.EMPTY) {
028            return "";
029        } else {
030            return toJson();
031        }
032    }
033
034    @Override
035    public String toJson() {
036        return writeJson(new StringBuilder()).toString();
037    }
038
039    @Override
040    public StringBuilder writeJson(StringBuilder target) {
041        return JsonRender.render(value, target, true);
042    }
043
044}
