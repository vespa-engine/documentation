002package com.yahoo.search.result;
003
004/**
005 * A class representing unset or undefined numeric values.
006 *
007 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
008 */
009@SuppressWarnings("serial")
010public final class NanNumber extends Number {
011    public static final NanNumber NaN = new NanNumber();
012
013    private NanNumber() {
014    }
015
016    @Override
017    public double doubleValue() {
018        return Double.NaN;
019    }
020
021    @Override
022    public float floatValue() {
023        return Float.NaN;
024    }
025
026    @Override
027    public int intValue() {
028        return 0;
029    }
030
031    @Override
032    public long longValue() {
033        return 0L;
034    }
035
036    @Override
037    public String toString() {
038        return "";
039    }
040
041}
