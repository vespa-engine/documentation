018/**
019 * <p>A search hit. The identifier of the hit is the uri
020 * (the uri is immutable once set).
021 * If two hits have the same uri they are equal per definition.
022 * Hits are naturally ordered by decreasing relevance.
023 * Note that this definition of equals and natural ordering is inconsistent.</p>
024 *
025 * <p>Hits may be of the <i>meta</i> type, meaning that they contain some information
026 * about the query or result which does not represent a particular piece of matched
027 * content. Meta hits are not counted in the hit count of the result, and should
028 * usually never be filtered out.</p>
029 *
030 * <p>Some hit sources may produce hits which are not <i>filled</i>. A non-filled
031 * hit may miss some or all of its property values. To fill those,
032 * {@link com.yahoo.search.Searcher#fill fill} must be called on the search chain by the searcher
033 * which requires those properties. This mechanism allows initial filtering to be
034 * done of a lightweight version of the hits, which is cheaper if a significant
035 * number of hits are filtered out.</p>
036 *
037 * @author bratseth
038 */
039public class Hit extends ListenableFreezableClass implements Data, Comparable<Hit>, Cloneable {

121    public static String stripCharacter(char strip, String toStripFrom) {
122        StringBuilder builder = null;
123
124        int lastBadChar = 0;
125        for (int i = 0; i < toStripFrom.length(); i++) {
126            if (toStripFrom.charAt(i) == strip) {
127                if (builder == null) {
128                    builder = new StringBuilder(toStripFrom.length());
129                }
130
131                builder.append(toStripFrom, lastBadChar, i);
132                lastBadChar = i + 1;
133            }
134        }
135
136        if (builder == null) {
137            return toStripFrom;
138        } else {
139            if (lastBadChar < toStripFrom.length()) {
140                builder.append(toStripFrom, lastBadChar, toStripFrom.length());
141            }
142
143            return builder.toString();
144        }
145    }

397        if (filled == null || filled.size() == 0) {

401            filled = new HashSet<>(filled);
402            unmodifiableFilled = Collections.unmodifiableSet(filled);
403
404            filled.add(summaryClass);
405        } else {
406            filled.add(summaryClass);
407        }
408    }

480    public HitField buildHitField(String key, boolean forceNoPreTokenize, boolean forceStringHandling) {
481        Object o = getField(key);
482        if (o == null) {
483            return null;
484        }
485
486        if (o instanceof HitField) {
487            return (HitField) o;
488        }
489
490        HitField h;
491        if (forceNoPreTokenize) {
492            if (o instanceof XMLString && !forceStringHandling) {
493                h = new HitField(key, (XMLString) o, false);
494            } else {
495                h = new HitField(key, o.toString(), false);
496            }
497        } else {
498            if (o instanceof XMLString && !forceStringHandling) {
499                h = new HitField(key, (XMLString) o);
500            } else {
501                h = new HitField(key, o.toString());
502            }
503        }
504        h.setOriginal(o);
505        getFieldMap().put(key, h);
506        return h;
507    }

559    /** Compares this hit to another hit */
560    public int compareTo(Hit other) {
561        // higher relevance is better
562        int result = other.getRelevance().compareTo(getRelevance());
563        if (result != 0) {
564            return result;
565        }
566        // lower addnumber is better
567        result = this.getAddNumber() - other.getAddNumber();
568        if (result != 0) {
569            return result;
570        }
571
572        // if all else fails, compare URIs (alphabetically)
573        if (this.getId() == null && other.getId() == null) {
574            return 0;
575        } else if (other.getId() == null) {
576            return -1;
577        } else if (this.getId() == null) {
578            return 1;
579        } else {
580            return this.getId().compareTo(other.getId());
581        }
582    }

688    /**
689     * Returns a field of this hit XML escaped and without token
690     * delimiters.
691     *
692     * @return a field of this hit, or null if the property is not set
693     */
694    public String getFieldXML(String key) {
695        Object p = getField(key);
696
697        if (p == null) {
698            return null;
699        } else if (p instanceof HitField) {
700            HitField hf = (HitField) p;
701
702            return hf.quotedContent(false);
703        } else if (p instanceof StructuredData) {
704            return p.toString();
705        } else if (p instanceof XMLString || p instanceof JSONString) {
706            return p.toString();
707        } else {
708            return XML.xmlEscape(p.toString(), false, '\u001f');
709        }

713    /**
714     * @return a field without bolding markup
715     */
716    public String getUnboldedField(String key, boolean escape) {
717        Object p = getField(key);
718
719        if (p == null) {
720            return null;
721        } else if (p instanceof HitField) {
722            return ((HitField) p).bareContent(escape, false);
723        } else if (p instanceof StructuredData) {
724            return p.toString();
725        } else if (p instanceof XMLString || p instanceof JSONString) {
726            return p.toString();
727        } else if (escape) {
728            return XML.xmlEscape(p.toString(), false, '\u001f');
729        } else {
730            return stripCharacter('\u001F', p.toString());
731        }
732    }

735     * set meta data describing how a given searcher should treat this hit.
736     * It is currently recommended that the invoker == searcher.
737     * <b>Internal. Do not use!</b>
738     */
739    public void setSearcherSpecificMetaData(Searcher searcher, Object data) {
740        if (searcherSpecificMetaData == null) {
741            searcherSpecificMetaData = Collections.singletonMap(searcher, data);
742        } else {
743            if (searcherSpecificMetaData.size() == 1) {
744                Object tmp = searcherSpecificMetaData.get(searcher);
745                if (tmp != null) {
746                    searcherSpecificMetaData = Collections.singletonMap(searcher, data);
747                } else {
748                    searcherSpecificMetaData = new TreeMap<>(searcherSpecificMetaData);
749                    searcherSpecificMetaData.put(searcher, data);
750                }
751            } else {
752                searcherSpecificMetaData.put(searcher, data);
753            }
754        }
755    }
