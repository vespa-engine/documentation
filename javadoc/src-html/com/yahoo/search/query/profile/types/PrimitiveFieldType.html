002package com.yahoo.search.query.profile.types;
003
004import com.yahoo.search.query.profile.QueryProfileRegistry;
005import com.yahoo.search.query.profile.compiled.CompiledQueryProfileRegistry;
006
007import static com.yahoo.text.Lowercase.toLowerCase;
008
009/**
010 * Represents a query field type which is a primitive - String, Integer, Float, Double or Long.
011 *
012 * @author bratseth
013 */
014@SuppressWarnings("rawtypes")
015public class PrimitiveFieldType extends FieldType {
016
017    private Class primitiveClass;
018
019    PrimitiveFieldType(Class primitiveClass) {
020        this.primitiveClass=primitiveClass;
021    }
022
023    public @Override Class getValueClass() { return primitiveClass; }
024
025    public @Override String stringValue() {
026        return toLowerCase(primitiveClass.getSimpleName());
027    }
028
029    public @Override String toString() { return "field type " + stringValue(); }
030
031    public @Override String toInstanceDescription() {
032        return toLowerCase(primitiveClass.getSimpleName());
033    }
034
035    @Override
036    public Object convertFrom(Object object, CompiledQueryProfileRegistry registry) {
037        return convertFrom(object, (QueryProfileRegistry)null);
038    }
039
040    public @Override Object convertFrom(Object object, QueryProfileRegistry registry) {
041        if (primitiveClass == object.getClass()) return object;
042
043        if (object.getClass() == String.class) return convertFromString((String)object);
044        if (object instanceof Number) return convertFromNumber((Number)object);
045
046        return null;
047    }
048
049    private Object convertFromString(String string) {
050        try {
051            if (primitiveClass==Integer.class) return Integer.valueOf(string);
052            if (primitiveClass==Double.class) return Double.valueOf(string);
053            if (primitiveClass==Float.class) return Float.valueOf(string);
054            if (primitiveClass==Long.class) return Long.valueOf(string);
055            if (primitiveClass==Boolean.class) return Boolean.valueOf(string);
056        }
057        catch (NumberFormatException e) {
058            return null; // Handled in caller
059        }
060        throw new RuntimeException("Programming error");
061    }
062
063    private Object convertFromNumber(Number number) {
064        if (primitiveClass==Integer.class) return number.intValue();
065        if (primitiveClass==Double.class) return number.doubleValue();
066        if (primitiveClass==Float.class) return number.floatValue();
067        if (primitiveClass==Long.class) return number.longValue();
068        if (primitiveClass==String.class) return String.valueOf(number);
069        throw new RuntimeException("Programming error: Input type is " + number.getClass() +
070                                   " primitiveClass is " + primitiveClass);
071    }
072
073    @Override
074    public int hashCode() {
075        return primitiveClass.hashCode();
076    }
077
078    @Override
079    public boolean equals(Object o) {
080        if (o == this) return true;
081        if ( ! (o instanceof PrimitiveFieldType)) return false;
082        PrimitiveFieldType other = (PrimitiveFieldType)o;
083        return other.primitiveClass.equals(this.primitiveClass);
084    }
085
086}
