017/**
018 * This rewriter would retrieve spell corrected query from QLAS and
019 * add it to the original query tree as equiv<br>
020 * - Features:<br>
021 *   RewritesAsEquiv flag: add rewrites to original query as equiv
022 *
023 * @author Karen Sze Wing Lee
024 */
025@After("QLAS")
026@Provides("MisspellRewriter")
027public class MisspellRewriter extends QueryRewriteSearcher {
028
029    // Flag for skipping this rewriter if the query has been rewritten
030    private final boolean SKIP_REWRITER_IF_REWRITTEN = false;
031
032    // Name of the rewriter
033    public static final String REWRITER_NAME = "MisspellRewriter";
034
035    private Logger logger = Logger.getLogger(MisspellRewriter.class.getName());
036
037    /**
038     * Constructor for MisspellRewriter
039     */
040    @Inject
041    public MisspellRewriter(ComponentId id) {
042        super(id);
043    }
044
045    /**
046     * Constructor for MisspellRewriter unit test
047     */
048    public MisspellRewriter() {
049        super();
050    }
051
052    /**
053     * Instance creation time config loading besides FSA.
054     * Empty for this rewriter
055     */
056    public boolean configure(FileAcquirer fileAcquirer,
057                             RewritesConfig config,
058                             HashMap<String, File> fileList) {
059        return true;
060    }
061
062    /**
063     * Main logic of rewriter<br>
064     * - Retrieve spell corrected query from QLAS<br>
065     * - Add spell corrected query as equiv
066     */
067    public HashMap<String, Object> rewrite(Query query,
068                                           String dictKey) throws RuntimeException {
069
070        Boolean rewritten = false;
071
072        HashMap<String, Object> result = new HashMap<>();
073        result.put(RewriterConstants.REWRITTEN, rewritten);
074        result.put(RewriterConstants.DICT_KEY, dictKey);
075
076        RewriterUtils.log(logger, query,
077                         "In MisspellRewriter");
078
079        // Retrieve flags for enabling the features
080        String qssRw = getQPConfig(query, RewriterConstants.QSS_RW);
081        String qssSugg = getQPConfig(query, RewriterConstants.QSS_SUGG);
082
083        boolean isQSSRw = false;
084        boolean isQSSSugg = false;
085
086        if(qssRw!=null) {
087            isQSSRw = qssRw.equalsIgnoreCase("true");
088        }
089        if(qssSugg!=null) {
090            isQSSSugg = qssSugg.equalsIgnoreCase("true");
091        }
092
093        // Rewrite is not enabled
094        if(!isQSSRw && !isQSSSugg) {
095            return result;
096        }
097
098        // Retrieve spell corrected query from QLAS
099        String rewrites = RewriterUtils.getSpellCorrected(query, isQSSRw, isQSSSugg);
100
101        // No rewrites
102        if(rewrites==null) {
103            RewriterUtils.log(logger, query, "No rewrite is retrieved");
104            return result;
105        } else {
106            RewriterUtils.log(logger, query, "Retrieved spell corrected query: " +
107                              rewrites);
108        }
109
110        // Adding rewrite to the query tree
111        query = RewriterFeatures.addRewritesAsEquiv(query, dictKey, rewrites, false, 0);
112
113        rewritten = true;
114        RewriterUtils.log(logger, query, "MisspellRewriter final query: " +
115                          query.toDetailString());
116
117        result.put(RewriterConstants.REWRITTEN, rewritten);
118        result.put(RewriterConstants.DICT_KEY, rewrites);
119
120        return result;
121    }
122
123    /**
124     * Get the flag which specifies whether this rewriter
125     * should be skipped if the query has been rewritten
126     *
127     * @return true if rewriter should be skipped, false
128     *         otherwise
129     */
130    public boolean getSkipRewriterIfRewritten() {
131        return SKIP_REWRITER_IF_REWRITTEN;
132    }
133
134   /**
135    * Get the name of the rewriter
136    *
137    * @return Name of the rewriter
138    */
139   public String getRewriterName() {
140       return REWRITER_NAME;
141   }
142
143   /**
144    * Get default FSA dictionary names
145    *
146    * @return Pair of FSA dictionary name and filename
147    */
148   public HashMap<String, String> getDefaultFSAs() {
149       return null;
150   }
151}
