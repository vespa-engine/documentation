002package com.yahoo.search.pagetemplates.result;
003
004import com.yahoo.search.pagetemplates.model.Renderer;
005import com.yahoo.search.pagetemplates.model.Source;
006import com.yahoo.search.result.HitGroup;
007
008import java.util.ArrayList;
009import java.util.List;
010
011/**
012 * A hit group corresponding to a section - contains some additional information
013 * in proper getters and setters which is used during rendering.
014 *
015 * @author bratseth
016 */
017public class SectionHitGroup extends HitGroup {
018
019    private static final long serialVersionUID = -9048845836777953538L;
020    private List<Source> sources=new ArrayList<>(0);
021    private List<Renderer> renderers=new ArrayList<>(0);
022    private final String displayId;
023
024    private boolean leaf=false;
025
026    public SectionHitGroup(String id) {
027        super(id);
028        if (id.startsWith("section:section_"))
029            displayId=null; // Don't display section ids when not named explicitly
030        else
031            displayId=id;
032        types().add("section");
033    }
034
035    @Override
036    public String getDisplayId() { return displayId; }
037
038    /**
039     * Returns the live, modifiable list of sources which are not fetched by the framework but should
040     * instead be included in the result
041     */
042    public List<Source> sources() { return sources; }
043
044    /** Returns the live, modifiable list of renderers in this section */
045    public List<Renderer> renderers() { return renderers; }
046
047    /** Returns whether this is a leaf section containing no subsections */
048    public boolean isLeaf() { return leaf; }
049
050    public void setLeaf(boolean leaf) { this.leaf=leaf; }
051
052}
