002package com.yahoo.search.grouping.result;
003
004import static com.yahoo.text.Lowercase.toLowerCase;
005
006/**
007 * This abstract class is used in {@link Group} instances where the identifying expression evaluated to a {@link
008 * com.yahoo.search.grouping.request.BucketValue}. The range is inclusive-from and exclusive-to.
009 *
010 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
011 */
012public abstract class BucketGroupId<T> extends GroupId {
013
014    private final T from;
015    private final T to;
016
017    /**
018     * Constructs a new instance of this class.
019     *
020     * @param type The type of this id's value.
021     * @param from The inclusive-from of the range.
022     * @param to   The exclusive-to of the range.
023     */
024    public BucketGroupId(String type, T from, T to) {
025        this(type, from, String.valueOf(from), to, String.valueOf(to));
026    }
027
028    /**
029     * Constructs a new instance of this class.
030     *
031     * @param type      The type of this id's value.
032     * @param from      The inclusive-from of the range.
033     * @param fromImage The String representation of the <tt>from</tt> argument.
034     * @param to        The exclusive-to of the range.
035     * @param toImage   The String representation of the <tt>to</tt> argument.
036     */
037    public BucketGroupId(String type, T from, String fromImage, T to, String toImage) {
038        super(type, fromImage, toImage);
039        this.from = from;
040        this.to = to;
041    }
042
043    /**
044     * Returns the inclusive-from of the value range.
045     *
046     * @return The from-value.
047     */
048    public T getFrom() {
049        return from;
050    }
051
052    /**
053     * Returns the exclusive-to of the value range.
054     *
055     * @return The to-value.
056     */
057    public T getTo() {
058        return to;
059    }
060}
