002package com.yahoo.search.grouping.request;
003
004/**
005 * This class represents a {@link Long} bucket in a {@link PredefinedFunction}.
006 *
007 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
008 */
009public class LongBucket extends BucketValue {
010
011    /**
012     * Gives the next distinct long value.
013     *
014     * @param value the base value.
015     * @return the nextt value.
016     */
017    public static LongValue nextValue(LongValue value) {
018        long v = value.getValue();
019        return new LongValue(v < Long.MAX_VALUE ? v + 1 : v);
020    }
021
022    /**
023     * Constructs a new instance of this class.
024     *
025     * @param from          The from-value to assign to this.
026     * @param to            The to-value to assign to this.
027     */
028    public LongBucket(long from, long to) {
029        super(new LongValue(from), new LongValue(to));
030    }
031
032    /**
033     * Constructs a new instance of this class.
034     *
035     * @param from          The from-value to assign to this.
036     * @param to            The to-value to assign to this.
037     */
038    @SuppressWarnings("rawtypes")
039    public LongBucket(ConstantValue from, ConstantValue to) {
040        super(from, to);
041    }
042}
