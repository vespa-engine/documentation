004/**
005 * An exception to signal abort current action, as the container is overloaded.
006 * Just unroll state as cheaply as possible.
007 *
008 * <p>
009 * The contract of OverloadException (for Jetty) is:
010 * </p>
011 *
012 * <ol>
013 * <li>You must set the response yourself first, or you'll get 500 internal
014 * server error.</li>
015 * <li>You must throw it from handleRequest synchronously.</li>
016 * </ol>
017 *
018 * @author <a href="mailto:steinar@yahoo-inc.com">Steinar Knutsen</a>
019 */
020public class OverloadException extends RuntimeException {
021    public OverloadException(String message, Throwable cause) {
022        super(message, cause);
023    }
024}
