002package com.yahoo.jdisc.handler;
003
004import java.nio.ByteBuffer;
005import java.util.LinkedList;
006import java.util.List;
007import java.util.Objects;
008
009/**
010 * <p>This class implements an unlimited, non-blocking content queue. All {@link ContentChannel} methods are implemented
011 * by pushing to a thread-safe internal queue. All of the queued calls are forwarded to another ContentChannel when
012 * {@link #connectTo(ContentChannel)} is called. Once connected, this class becomes a non-buffering proxy for the
013 * connected ContentChannel.</p>
014 *
015 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
016 */
017public final class BufferedContentChannel implements ContentChannel {
018
019    private final Object lock = new Object();
020    private List<Entry> queue = new LinkedList<>();
021    private ContentChannel content = null;
022    private boolean closed = false;
023    private CompletionHandler closeCompletion = null;
024
025    /**
026     * <p>Connects this BufferedContentChannel to a ContentChannel. First, this method forwards all queued calls to the
027     * connected ContentChannel. Once this method has been called, all future calls to {@link #write(ByteBuffer,
028     * CompletionHandler)} and {@link #close(CompletionHandler)} are synchronously forwarded to the connected
029     * ContentChannel.</p>
030     *
031     * @param content The ContentChannel to connect to.
032     * @throws NullPointerException  If the <em>content</em> argument is null.
033     * @throws IllegalStateException If another ContentChannel has already been connected.
034     */
035    public void connectTo(ContentChannel content) {
036        Objects.requireNonNull(content, "content");
037        boolean closed;
038        List<Entry> queue;
039        synchronized (lock) {
040            if (this.content != null || this.queue == null) {
041                throw new IllegalStateException();
042            }
043            closed = this.closed;
044            queue = this.queue;
045            this.queue = null;
046        }
047        for (Entry entry : queue) {
048            content.write(entry.buf, entry.handler);
049        }
050        if (closed) {
051            content.close(closeCompletion);
052        }
053        synchronized (lock) {
054            this.content = content;
055            lock.notifyAll();
056        }
057    }
058
059    /**
060     * <p>Returns whether or not {@link #connectTo(ContentChannel)} has been called. Even if this method returns false,
061     * calling {@link #connectTo(ContentChannel)} might still throw an IllegalStateException if there is a race.</p>
062     *
063     * @return True if {@link #connectTo(ContentChannel)} has been called.
064     */
065    public boolean isConnected() {
066        synchronized (lock) {
067            return content != null;
068        }
069    }
070
071    /**
072     * <p>Creates a {@link ReadableContentChannel} and {@link #connectTo(ContentChannel) connects} to it. </p>
073     *
074     * @return The new ReadableContentChannel that this connected to.
075     */
076    public ReadableContentChannel toReadable() {
077        ReadableContentChannel ret = new ReadableContentChannel();
078        connectTo(ret);
079        return ret;
080    }
081
082    /**
083     * <p>Creates a {@link ContentInputStream} and {@link #connectTo(ContentChannel) connects} to its internal
084     * ContentChannel.</p>
085     *
086     * @return The new ContentInputStream that this connected to.
087     */
088    public ContentInputStream toStream() {
089        return toReadable().toStream();
090    }
091
092    @Override
093    public void write(ByteBuffer buf, CompletionHandler handler) {
094        ContentChannel content;
095        synchronized (lock) {
096            if (closed) {
097                throw new IllegalStateException();
098            }
099            if (queue != null) {
100                queue.add(new Entry(buf, handler));
101                return;
102            }
103            try {
104                while (this.content == null) {
105                    lock.wait(); // waiting for connecTo()
106                }
107            } catch (InterruptedException e) {
108                throw new IllegalStateException(e);
109            }
110            if (closed) {
111                throw new IllegalStateException();
112            }
113            content = this.content;
114        }
115        content.write(buf, handler);
116    }
117
118    @Override
119    public void close(CompletionHandler handler) {
120        ContentChannel content;
121        synchronized (lock) {
122            if (closed) {
123                throw new IllegalStateException();
124            }
125            if (queue != null) {
126                closed = true;
127                closeCompletion = handler;
128                return;
129            }
130            try {
131                while (this.content == null) {
132                    lock.wait(); // waiting for connecTo()
133                }
134            } catch (InterruptedException e) {
135                throw new IllegalStateException(e);
136            }
137            if (closed) {
138                throw new IllegalStateException();
139            }
140            closed = true;
141            content = this.content;
142        }
143        content.close(handler);
144    }
145
146    private static class Entry {
147
148        final ByteBuffer buf;
149        final CompletionHandler handler;
150
151        Entry(ByteBuffer buf, CompletionHandler handler) {
152            this.handler = handler;
153            this.buf = buf;
154        }
155    }
156}
