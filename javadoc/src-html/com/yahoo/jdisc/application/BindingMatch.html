002package com.yahoo.jdisc.application;
003
004import java.net.URI;
005import java.util.Objects;
006
007/**
008 * <p>This class holds the result of a {@link BindingSet#match(URI)} operation. It contains methods to inspect the
009 * groups captured during matching, where a <em>group</em> is defined as a sequence of characters matches by a wildcard
010 * in the {@link UriPattern}, and to retrieve the matched target.</p>
011 *
012 * @param <T> The class of the target.
013 */
014public class BindingMatch<T> {
015
016    private final UriPattern.Match match;
017    private final T target;
018
019    /**
020     * <p>Constructs a new instance of this class.</p>
021     *
022     * @param match  The match information for this instance.
023     * @param target The target of this match.
024     * @throws NullPointerException If any argument is null.
025     */
026    public BindingMatch(UriPattern.Match match, T target) {
027        Objects.requireNonNull(match, "match");
028        Objects.requireNonNull(target, "target");
029        this.match = match;
030        this.target = target;
031    }
032
033    /**
034     * <p>Returns the number of captured groups of this match. Any non-negative integer smaller than the value returned
035     * by this method is a valid group index for this match.</p>
036     *
037     * @return The number of captured groups.
038     */
039    public int groupCount() {
040        return match.groupCount();
041    }
042
043    /**
044     * <p>Returns the input subsequence captured by the given group by this match. Groups are indexed from left to
045     * right, starting at zero. Note that some groups may match an empty string, in which case this method returns the
046     * empty string. This method never returns null.</p>
047     *
048     * @param idx The index of the group to return.
049     * @return The (possibly empty) substring captured by the group during matching, never <tt>null</tt>.
050     * @throws IndexOutOfBoundsException If there is no group in the match with the given index.
051     */
052    public String group(int idx) {
053        return match.group(idx);
054    }
055
056    /**
057     * <p>Returns the matched target.</p>
058     *
059     * @return The matched target.
060     */
061    public T target() {
062        return target;
063    }
064}
