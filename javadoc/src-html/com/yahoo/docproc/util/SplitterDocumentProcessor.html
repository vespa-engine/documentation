002package com.yahoo.docproc.util;
003
004import com.yahoo.document.DocumentOperation;
005import com.yahoo.document.DocumentPut;
006import com.yahoo.document.config.DocumentmanagerConfig;
007import com.yahoo.config.docproc.SplitterJoinerDocumentProcessorConfig;
008import com.yahoo.docproc.DocumentProcessor;
009import com.yahoo.docproc.Processing;
010import com.yahoo.document.ArrayDataType;
011import com.yahoo.document.Document;
012import com.yahoo.document.DocumentType;
013import com.yahoo.document.DocumentTypeManager;
014import com.yahoo.document.DocumentTypeManagerConfigurer;
015import com.yahoo.document.datatypes.Array;
016import com.yahoo.log.LogLevel;
017
018import java.util.logging.Logger;
019import java.util.stream.Collectors;
020
021/**
022 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
023 */
024public class SplitterDocumentProcessor extends DocumentProcessor {
025
026    private static Logger log = Logger.getLogger(SplitterDocumentProcessor.class.getName());
027    private String documentTypeName;
028    private String arrayFieldName;
029    private String contextFieldName;
030    DocumentTypeManager manager;
031
032    public SplitterDocumentProcessor(SplitterJoinerDocumentProcessorConfig cfg, DocumentmanagerConfig documentmanagerConfig) {
033        super();
034        this.documentTypeName = cfg.documentTypeName();
035        this.arrayFieldName = cfg.arrayFieldName();
036        this.contextFieldName = cfg.contextFieldName();
037        this.manager = DocumentTypeManagerConfigurer.configureNewManager(documentmanagerConfig);
038        validate(manager, documentTypeName, arrayFieldName);
039    }
040
041    @Override
042    public Progress process(Processing processing) {
043        if (processing.getDocumentOperations().size() != 1) {
044            //we were given more than one document, return
045            log.log(LogLevel.DEBUG, "More than one document given, returning. (Was given "
046                                    + processing.getDocumentOperations().size() + " documents).");
047            return Progress.DONE;
048        }
049
050        if (!doProcessOuterDocument(processing.getDocumentOperations().get(0), documentTypeName)) {
051            return Progress.DONE;
052        }
053
054        Document outerDoc = ((DocumentPut)processing.getDocumentOperations().get(0)).getDocument();;
055
056        Array<Document> innerDocuments = (Array<Document>) outerDoc.getFieldValue(arrayFieldName);
057        if (innerDocuments == null) {
058            //the document does not have the field, return
059            log.log(LogLevel.DEBUG, "The given Document does not have a field value for field "
060                                    + arrayFieldName + ", returning. (Was given " + outerDoc + ").");
061            return Progress.DONE;
062        }
063
064        if (innerDocuments.size() == 0) {
065            //the array is empty, return
066            log.log(LogLevel.DEBUG, "The given Document does not have any elements in array field "
067                                    + arrayFieldName + ", returning. (Was given " + outerDoc + ").");
068            return Progress.DONE;
069        }
070
071        split(processing, innerDocuments);
072        return Progress.DONE;
073    }
074
075    private void split(Processing processing, Array<Document> innerDocuments) {
076        processing.setVariable(contextFieldName, processing.getDocumentOperations().get(0));
077        processing.getDocumentOperations().clear();
078        processing.getDocumentOperations().addAll(innerDocuments.stream()
079                .map(DocumentPut::new)
080                .collect(Collectors.toList()));
081
082        innerDocuments.clear();
083    }
084
085
086    static void validate(DocumentTypeManager manager, String documentTypeName, String arrayFieldName) {
087        DocumentType docType = manager.getDocumentType(documentTypeName);
088
089        if (docType == null) {
090            //the document type does not exist, return
091            throw new IllegalStateException("The document type " + documentTypeName + " is not deployed.");
092        }
093
094        if (docType.getField(arrayFieldName) == null) {
095            //the document type does not have the field, return
096            throw new IllegalStateException("The document type " + documentTypeName
097                                            + " does not have a field named " + arrayFieldName + ".");
098        }
099
100        if (!(docType.getField(arrayFieldName).getDataType() instanceof ArrayDataType)) {
101            //the data type of the field is wrong, return
102            throw new IllegalStateException("The data type of the field named "
103                                            + arrayFieldName + " in document type " + documentTypeName
104                                            + " is not an array type");
105        }
106
107        ArrayDataType fieldDataType = (ArrayDataType) docType.getField(arrayFieldName).getDataType();
108
109        if (!(fieldDataType.getNestedType() instanceof DocumentType)) {
110            //the subtype of tye array data type of the field is wrong, return
111            throw new IllegalStateException("The data type of the field named "
112                                            + arrayFieldName + " in document type " + documentTypeName
113                                            + " is not an array of Document.");
114        }
115    }
116
117    static boolean doProcessOuterDocument(Object o, String documentTypeName) {
118        if ( ! (o instanceof DocumentOperation)) {
119            if (log.isLoggable(LogLevel.DEBUG)) {
120                log.log(LogLevel.DEBUG, o + " is not a DocumentOperation.");
121            }
122            return false;
123        }
124
125        DocumentOperation outerDocOp = (DocumentOperation)o;
126        if ( ! (outerDocOp instanceof DocumentPut)) {
127            //this is not a put, return
128            if (log.isLoggable(LogLevel.DEBUG)) {
129                log.log(LogLevel.DEBUG, "Given DocumentOperation is not a DocumentPut, returning. (Was given "
130                                        + outerDocOp + ").");
131            }
132            return false;
133        }
134
135        Document outerDoc = ((DocumentPut) outerDocOp).getDocument();
136        DocumentType type = outerDoc.getDataType();
137        if (!type.getName().equalsIgnoreCase(documentTypeName)) {
138            //this is not the right document type
139            if (log.isLoggable(LogLevel.DEBUG)) {
140                log.log(LogLevel.DEBUG, "Given Document is of wrong type, returning. (Was given " + outerDoc + ").");
141            }
142            return false;
143        }
144        return true;
145    }
