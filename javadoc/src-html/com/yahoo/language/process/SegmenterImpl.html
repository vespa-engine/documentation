002package com.yahoo.language.process;
003
004import com.yahoo.language.Language;
005
006import java.util.ArrayList;
007import java.util.List;
008
009/**
010 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen Hult</a>
011 */
012public class SegmenterImpl implements Segmenter {
013
014    private final Tokenizer tokenizer;
015
016    public SegmenterImpl(Tokenizer tokenizer) {
017        this.tokenizer = tokenizer;
018    }
019
020    @Override
021    public List<String> segment(String input, Language language) {
022        List<String> segments = new ArrayList<>();
023        for (Token token : tokenizer.tokenize(input, language, StemMode.NONE, false)) {
024            findSegments(token, segments);
025        }
026        if (segments.isEmpty()) {
027            segments.add(input); // no segments, return original string
028        }
029        return segments;
030    }
031
032    private void findSegments(Token token, List<String> out) {
033        int len;
034        if (token.isSpecialToken() || (len = token.getNumComponents()) == 0) {
035            if (token.isIndexable()) {
036                out.add(token.getOrig());
037            }
038        } else {
039            for (int i = 0; i < len; ++i) {
040                findSegments(token.getComponent(i), out);
041            }
042        }
043    }
044
045}
