002package com.yahoo.document.datatypes;
003
004import com.yahoo.document.DataType;
005import com.yahoo.document.Field;
006import com.yahoo.document.PrimitiveDataType;
007import com.yahoo.document.predicate.Predicate;
008import com.yahoo.document.serialization.FieldReader;
009import com.yahoo.document.serialization.FieldWriter;
010import com.yahoo.document.serialization.XmlStream;
011
012import java.util.Objects;
013
014/**
015 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen Hult</a>
016 */
017public class PredicateFieldValue extends FieldValue {
018
019    private Predicate predicate;
020
021    public PredicateFieldValue() {
022        this((Predicate)null);
023    }
024
025    public PredicateFieldValue(Predicate predicate) {
026        this.predicate = predicate;
027    }
028
029    public PredicateFieldValue(String predicateString) {
030        this(Predicate.fromString(predicateString));
031    }
032
033    public Predicate getPredicate() {
034        return predicate;
035    }
036
037    public PredicateFieldValue setPredicate(Predicate predicate) {
038        this.predicate = predicate;
039        return this;
040    }
041
042    @Override
043    public DataType getDataType() {
044        return DataType.PREDICATE;
045    }
046
047    @Override
048    public void printXml(XmlStream xml) {
049        if (predicate == null) {
050            return;
051        }
052        xml.addContent(predicate.toString());
053    }
054
055    @Override
056    public void clear() {
057        predicate = null;
058    }
059
060    @Override
061    public void assign(Object o) {
062        if (o == null) {
063            predicate = null;
064        } else if (o instanceof Predicate) {
065            predicate = (Predicate)o;
066        } else if (o instanceof PredicateFieldValue) {
067            predicate = ((PredicateFieldValue)o).predicate;
068        } else {
069            throw new IllegalArgumentException("Expected " + getClass().getName() + ", got " +
070                                               o.getClass().getName() + ".");
071        }
072    }
073
074    @Override
075    public void serialize(Field field, FieldWriter writer) {
076        writer.write(field, this);
077    }
078
079    @Override
080    public void deserialize(Field field, FieldReader reader) {
081        reader.read(field, this);
082    }
083
084    @Override
085    public Object getWrappedValue() {
086        return predicate;
087    }
088
089    @Override
090    public PredicateFieldValue clone() {
091        PredicateFieldValue obj = (PredicateFieldValue)super.clone();
092        if (predicate != null) {
093            try {
094                obj.predicate = predicate.clone();
095            } catch (CloneNotSupportedException e) {
096                throw new AssertionError(e);
097            }
098        }
099        return obj;
100    }
101
102    @Override
103    public int hashCode() {
104        return predicate != null ? predicate.hashCode() : 31;
105    }
106
107    @Override
108    public boolean equals(Object obj) {
109        if (obj == this) {
110            return true;
111        }
112        if (!(obj instanceof PredicateFieldValue)) {
113            return false;
114        }
115        PredicateFieldValue rhs = (PredicateFieldValue)obj;
116        if (!Objects.equals(predicate, rhs.predicate)) {
117            return false;
118        }
119        return true;
120    }
121
122    @Override
123    public String toString() {
124        return String.valueOf(predicate);
125    }
126
127    public static PrimitiveDataType.Factory getFactory() {
128        return new PrimitiveDataType.Factory() {
129
130            @Override
131            public FieldValue create() {
132                return new PredicateFieldValue();
133            }
134        };
135    }
136}
