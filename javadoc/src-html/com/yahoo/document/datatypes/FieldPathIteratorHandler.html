002package com.yahoo.document.datatypes;
003
004import java.util.HashMap;
005import java.util.Map;
006import java.util.TreeMap;
007
008/**
009 * @author <a href="mailto:thomasg@yahoo-inc.com">Thomas Gundersen</a>
010 */
011public abstract class FieldPathIteratorHandler {
012
013    public static class IndexValue {
014
015        private int index;
016        private FieldValue key;
017
018        public int getIndex() {
019            return index;
020        }
021
022        public FieldValue getKey() {
023            return key;
024        }
025
026        public IndexValue() {
027            index = -1;
028            key = null;
029        }
030
031        public IndexValue(int index) {
032            this.index = index;
033            key = null;
034        }
035
036        public IndexValue(FieldValue key) {
037            index = -1;
038            this.key = key;
039        }
040
041        public String toString() {
042            if (key != null) {
043                return key.toString();
044            } else {
045                return "" + index;
046            }
047        }
048
049        @Override
050        public boolean equals(Object o) {
051            IndexValue other = (IndexValue)o;
052
053            if (key != null) {
054                if (other.key != null && key.equals(other.key)) {
055                    return true;
056                }
057                return false;
058            }
059
060            return index == other.index;
061        }
062    };
063
064    public static class VariableMap extends TreeMap<String, IndexValue> {
065
066        @Override
067        public Object clone() {
068            Map<String, IndexValue> map = new VariableMap();
069            map.putAll(this);
070            return map;
071        }
072    }
073
074    private VariableMap variables = new VariableMap();
075
076    public void onPrimitive(FieldValue fv) {
077
078    }
079
080    public boolean onComplex(FieldValue fv) {
081        return true;
082    }
083
084    public ModificationStatus doModify(FieldValue fv) {
085        return ModificationStatus.NOT_MODIFIED;
086    }
087
088    public enum ModificationStatus {
089        MODIFIED, REMOVED, NOT_MODIFIED
090    }
091
092    public ModificationStatus modify(FieldValue fv) {
093        return doModify(fv);
094    }
095
096    public boolean createMissingPath() {
097        return false;
098    }
099
100    public VariableMap getVariables() {
101        return variables;
102    }
103}
