013/**
014 * A node in a {@link SpanNode} tree that can have a <strong>multiple</strong> trees of child nodes, each with its own probability.
015 * This class has quite a few convenience methods for accessing the <strong>first</strong> subtree.
016 *
017 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
018 * @see com.yahoo.document.annotation.SpanList
019 */
020public class AlternateSpanList extends SpanList {
021    public static final byte ID = 4;
022    private final List<Children> childTrees = new LinkedList<Children>();
023    private static final Comparator<Children> childComparator = new ProbabilityComparator();
024
025    /** Create a new AlternateSpanList instance, having a single subtree with probability 1.0. */
026    public AlternateSpanList() {
027        super((List<SpanNode>) null);
028        ensureAtLeastOneSubTree();
029    }
030
031    /*
032     * Deep-copies another AlternateSpanList.
033     *
034     * @param otherSpanList the instance to deep-copy.
035     */
036    public AlternateSpanList(AlternateSpanList otherSpanList) {
037        super((List<SpanNode>) null);
038        for (Children otherSubtree : otherSpanList.childTrees) {
039            //create our own subtree:
040            Children children = new Children(this);
041            //copy nodes:
042            for (SpanNode otherNode : otherSubtree.children()) {
043                if (otherNode instanceof Span) {
044                    children.add(new Span((Span) otherNode));
045                } else if (otherNode instanceof AlternateSpanList) {
046                    children.add(new AlternateSpanList((AlternateSpanList) otherNode));
047                } else if (otherNode instanceof SpanList) {
048                    children.add(new SpanList((SpanList) otherNode));
049                } else if (otherNode instanceof DummySpanNode) {
050                    children.add(otherNode);  //shouldn't really happen
051                } else {
052                    throw new IllegalStateException("Cannot create copy of " + otherNode + " with class "
053                                                    + ((otherNode == null) ? "null" : otherNode.getClass()));
054                }
055            }
056            //add this subtree to our subtrees:
057            childTrees.add(children);
058        }
059    }
060
061    public AlternateSpanList(SpanNodeReader reader) {
062        this();
063        reader.read(this);
064    }

066    private void ensureAtLeastOneSubTree() {
067        if (childTrees.isEmpty()) {
068            childTrees.add(new Children(getParent()));
069        }

185    /**
186     * Moves a child of this SpanList to another SpanList.
187     *
188     * @param i the index of the subtree to remove the node from
189     * @param node the node to move
190     * @param target the SpanList to add the node to
191     * @throws IllegalArgumentException if the given node is not a child of this SpanList
192     */
193    public void move(int i, SpanNode node, SpanList target) {
194        boolean removed = children(i).remove(node);
195        if (removed) {
196            //we found the node
197            node.setParent(null);
198            resetCachedFromAndTo();
199            target.add(node);
200        } else {
201            throw new IllegalArgumentException("Node " + node + " is not a child of this SpanList, cannot move.");
202        }
203    }

223    /**
224     * Moves a child of this SpanList to another SpanList.
225     *
226     * @param i the index of the subtree to remove the node from
227     * @param node the node to move
228     * @param target the SpanList to add the node to
229     * @param targetSubTree the index of the subtree of the given AlternateSpanList to add the node to
230     * @throws IllegalArgumentException if the given node is not a child of this SpanList
231     * @throws IndexOutOfBoundsException if the given index is out of range, or if the target subtree index is out of range
232     */
233    public void move(int i, SpanNode node, AlternateSpanList target, int targetSubTree) {
234        if (targetSubTree < 0 || targetSubTree >= target.getNumSubTrees()) {
235            throw new IndexOutOfBoundsException(target + " has no subtree at index " + targetSubTree);
236        }
237        boolean removed = children(i).remove(node);
238        if (removed) {
239            //we found the node
240            node.setParent(null);
241            resetCachedFromAndTo();
242            target.add(targetSubTree, node);
243        } else {
244            throw new IllegalArgumentException("Node " + node + " is not a child of this SpanList, cannot move.");
245        }
246    }

390    /**
391     * Removes the subtree at index i (both the subtree itself and its contents, which become invalidated).
392     * Note that if this AlternateSpanList has only one subtree and index 0 is given,
393     * a new empty subtree is automatically added, since an AlternateSpanList always has at least one subtree.
394     *
395     * @param i the index of the subtree to remove
396     * @return the subtree removed, if any (note: invalidated)
397     */
398    public List<SpanNode> removeChildren(int i) {
399        Children retval = childTrees.remove(i);
400        ensureAtLeastOneSubTree();
401        resetCachedFromAndTo();
402        if (retval != null) {
403            retval.setInvalid();
404            retval.setParent(null);
405            for (SpanNode node : retval.children()) {
406                node.setParent(null);
407            }
408            return retval.children();
409        }
410        return null;
411    }
412
413    /**
414     * Removes all subtrees (both the subtrees themselves and their contents, which become invalidated).
415     * Note that a new empty subtree is automatically added at index 0, since an AlternateSpanList always has at
416     * least one subtree.
417     */
418    public void removeChildren() {
419        for (Children ch : childTrees) {
420            ch.setInvalid();
421            ch.setParent(null);
422            ch.clearChildren();
423        }
424        childTrees.clear();
425        resetCachedFromAndTo();
426        ensureAtLeastOneSubTree();
427    }

440    /**
441     * Sets the subtree at index i.
442     *
443     * @param i           the index of where to set the subtree
444     * @param subtree     the subtree to set
445     * @param probability the probability to set
446     * @return the overwritten subtree, if any
447     */
448    public List<SpanNode> setChildren(int i, List<SpanNode> subtree, double probability) {
449        resetCachedFromAndTo();
450        if (childTrees.size() == 1 && i == 0) {
451            //replace the first subtree
452            Children sub = new Children(getParent(), subtree, probability);
453            Children retval = childTrees.set(i, sub);
454            if (retval == null) {
455                return null;
456            } else {
457                retval.setParent(null);
458                for (SpanNode node : retval.children()) {
459                    node.setParent(null);
460                }
461                return retval.children();
462            }
463        }
464        List<SpanNode> retval = removeChildren(i);
465        addChildren(i, subtree, probability);
466        return retval;
467    }

502    /**
503     * Returns the text covered by this span as given by subtree i, or null if subtree i is empty.
504     *
505     * @param i    the index of the subtree to use
506     * @param text the text to get a substring from
507     * @return the text covered by this span as given by subtree i, or null if subtree i is empty
508     */
509    public CharSequence getText(int i, CharSequence text) {
510        if (children(i).isEmpty()) {
511            return null;
512        }
513        StringBuilder str = new StringBuilder();
514        List<SpanNode> ch = children(i);
515        for (SpanNode node : ch) {
516            CharSequence childText = node.getText(text);
517            if (childText != null) {
518                str.append(node.getText(text));
519            }
520        }
521        return str;
522    }

544    /** Normalizes all probabilities between 0.0 (inclusive) and 1.0 (exclusive). */
545    public void normalizeProbabilities() {
546        double sum = 0.0;
547        for (Children c : childTrees) {
548            sum += c.getProbability();
549        }
550        double coeff = 1.0 / sum;
551
552        for (Children childTree : childTrees) {
553            double newProb = childTree.getProbability() * coeff;
554            childTree.setProbability(newProb);
555        }
556    }

559    /**
560     * Convenience method to add a span node to the child tree at index i. This is equivalent to calling
561     * <code>
562     * AlternateSpanList.children(i).add(node);
563     * </code>
564     *
565     * @param i    index
566     * @param node span node
567     */
568    public AlternateSpanList add(int i, SpanNode node) {
569        checkValidity(node, children(i));
570        node.setParent(this);
571        children(i).add(node);
572        return this;
573    }

601    private static class ProbabilityComparator implements Comparator<Children> {
602        @Override
603        public int compare(Children o1, Children o2) {
604            return Double.compare(o2.probability, o1.probability);  //note: opposite of natural ordering!
605        }
606    }
607
608    private class Children extends SpanList {
609        private double probability = 1.0;
610
611        private Children(SpanNodeParent parent) {
612            setParent(parent);
613        }
614
615        private Children(SpanNodeParent parent, List<SpanNode> children, double probability) {
616            super(children);
617            setParent(parent);
618            if (children != null) {
619                for (SpanNode node : children) {
620                    node.setParent(AlternateSpanList.this);
621                }
622            }
623            this.probability = probability;
624        }
625
626        public double getProbability() {
627            return probability;
628        }
629
630        public void setProbability(double probability) {
631            this.probability = probability;
632        }
633    }
