002package com.yahoo.document.annotation;
003
004import com.yahoo.document.DataType;
005import com.yahoo.document.datatypes.FieldValue;
006
007/**
008 * A data type describing a field value having a reference to an annotation of a given type.
009 *
010 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
011 */
012public class AnnotationReferenceDataType extends DataType {
013    private AnnotationType aType;
014
015    /**
016     * Creates an AnnotationReferenceDataType with a generated id.
017     *
018     * @param aType the annotation type that AnnotationRefs shall refer to.
019     */
020    public AnnotationReferenceDataType(AnnotationType aType) {
021        super("annotationreference<" + ((aType == null) ? "" : aType.getName()) + ">", 0);
022        setAnnotationType(aType);
023    }
024
025    /**
026     * Creates an AnnotationReferenceDataType with a given id.
027     *
028     * @param aType the annotation type that AnnotationRefs shall refer to.
029     * @param id    the id to use
030     */
031    public AnnotationReferenceDataType(AnnotationType aType, int id) {
032        super("annotationreference<" + ((aType == null) ? "" : aType.getName()) + ">", id);
033        this.aType = aType;
034    }
035
036    /**
037     * Creates an AnnotationReferenceDataType.&nbsp;WARNING! Do not use!
038     */
039    protected AnnotationReferenceDataType() {
040        super("annotationreference<>", 0);
041    }
042
043    private int createId() {
044        //TODO: This should be Java's hashCode(), since all other data types use it, and using something else here will probably lead to collisions
045        return getName().toLowerCase().hashCode();
046    }
047
048    @Override
049    public FieldValue createFieldValue() {
050        return new AnnotationReference(this);
051    }
052
053    @Override
054    public Class getValueClass() {
055        return AnnotationReference.class;
056    }
057
058    @Override
059    public boolean isValueCompatible(FieldValue value) {
060        if (!(value instanceof AnnotationReference)) {
061            return false;
062        }
063        AnnotationReference reference = (AnnotationReference) value;
064        if (equals(reference.getDataType())) {
065            return true;
066        }
067        return false;
068    }
069
070    /**
071     * Returns the annotation type of this AnnotationReferenceDataType.
072     *
073     * @return the annotation type of this AnnotationReferenceDataType.
074     */
075    public AnnotationType getAnnotationType() {
076        return aType;
077    }
078
079    /**
080     * Sets the annotation type that this AnnotationReferenceDataType points to.&nbsp;WARNING! Do not use.
081     * @param type the annotation type of this AnnotationReferenceDataType.
082     */
083    protected void setAnnotationType(AnnotationType type) {
084        this.aType = type;
085        setId(createId());
086    }
087}
