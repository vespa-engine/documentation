061    @SuppressWarnings("unchecked")
062    public SpanTree(SpanTree otherToCopy) {
063        name = otherToCopy.name;
064        setRoot(copySpan(otherToCopy.root));
065        List<Annotation> annotationsToCopy = new ArrayList<Annotation>(otherToCopy.getAnnotations());
066        List<Annotation> newAnnotations = new ArrayList<Annotation>(annotationsToCopy.size());
067
068        for (Annotation otherAnnotationToCopy : annotationsToCopy) {
069            newAnnotations.add(new Annotation(otherAnnotationToCopy));
070        }
071
072        IdentityHashMap<SpanNode, Integer> originalSpanNodes = getSpanNodes(otherToCopy);
073        List<SpanNode> copySpanNodes = getSpanNodes();
074
075        for (int i = 0; i < annotationsToCopy.size(); i++) {
076            Annotation originalAnnotation = annotationsToCopy.get(i);
077            if (!originalAnnotation.isSpanNodeValid()) {  //returns false also if spanNode is null!
078                continue;
079            }
080            Integer indexOfOriginalSpanNode = originalSpanNodes.get(originalAnnotation.getSpanNode());
081            if (indexOfOriginalSpanNode == null) {
082                throw new IllegalStateException("Could not clone tree, SpanNode of " + originalAnnotation + " not found.");
083            }
084            newAnnotations.get(i).setSpanNode(copySpanNodes.get(indexOfOriginalSpanNode));
085        }
086
087        IdentityHashMap<Annotation, Integer> originalAnnotations = getAnnotations(annotationsToCopy);
088
089        for (Annotation a : newAnnotations) {
090            if (!a.hasFieldValue()) {
091                continue;
092            }
093            setCorrectAnnotationReference(a.getFieldValue(), originalAnnotations, newAnnotations);
094        }
095
096        for (Annotation a : newAnnotations) {
097            annotate(a);
098        }
099        for (IndexKey key : otherToCopy.getCurrentIndexes()) {
100            createIndex(key);
101        }
102    }

104    private void setCorrectAnnotationReference(FieldValue value, IdentityHashMap<Annotation, Integer> originalAnnotations, List<Annotation> newAnnotations) {
105        if (value == null) {
106            return;
107        }
108
109        if (value.getDataType() instanceof AnnotationReferenceDataType) {
110            AnnotationReference ref = (AnnotationReference) value;
111            if (ref.getReference() == null) {
112                return;
113            }
114            Integer referenceIndex = originalAnnotations.get(ref.getReference());
115            if (referenceIndex == null) {
116                throw new IllegalStateException("Cannot find Annotation pointed to by " + ref);
117            }
118            try {
119                Annotation newReference = newAnnotations.get(referenceIndex);
120                ref.setReference(newReference);
121            } catch (IndexOutOfBoundsException ioobe) {
122                throw new IllegalStateException("Cannot find Annotation pointed to by " + ref, ioobe);
123            }
124        } else if (value.getDataType() instanceof StructuredDataType) {
125            setCorrectAnnotationReference((StructuredFieldValue) value, originalAnnotations, newAnnotations);
126        } else if (value.getDataType() instanceof CollectionDataType) {
127            setCorrectAnnotationReference((CollectionFieldValue) value, originalAnnotations, newAnnotations);
128        } else if (value.getDataType() instanceof MapDataType) {
129            setCorrectAnnotationReference((MapFieldValue) value, originalAnnotations, newAnnotations);
130        }
131    }
132
133    private void setCorrectAnnotationReference(StructuredFieldValue struct, IdentityHashMap<Annotation, Integer> originalAnnotations, List<Annotation> newAnnotations) {
134        for (Field f : struct.getDataType().getFields()) {
135            setCorrectAnnotationReference(struct.getFieldValue(f), originalAnnotations, newAnnotations);
136        }
137    }

139    private void setCorrectAnnotationReference(CollectionFieldValue collection, IdentityHashMap<Annotation, Integer> originalAnnotations, List<Annotation> newAnnotations) {
140        Iterator it = collection.fieldValueIterator();
141        while (it.hasNext()) {
142            setCorrectAnnotationReference((FieldValue) it.next(), originalAnnotations, newAnnotations);
143        }
144    }

146    private void setCorrectAnnotationReference(MapFieldValue map, IdentityHashMap<Annotation, Integer> originalAnnotations, List<Annotation> newAnnotations) {
147        for (Object o : map.values()) {
148            setCorrectAnnotationReference((FieldValue) o, originalAnnotations, newAnnotations);
149        }
150    }

152    private IdentityHashMap<Annotation, Integer> getAnnotations(List<Annotation> annotationsToCopy) {
153        IdentityHashMap<Annotation, Integer> map = new IdentityHashMap<Annotation, Integer>();
154        for (int i = 0; i < annotationsToCopy.size(); i++) {
155            map.put(annotationsToCopy.get(i), i);
156        }
157        return map;
158    }

161    private List<SpanNode> getSpanNodes() {
162        ArrayList<SpanNode> nodes = new ArrayList<SpanNode>();
163        nodes.add(root);
164        Iterator<SpanNode> it = root.childIteratorRecursive();
165        while (it.hasNext()) {
166            nodes.add(it.next());
167        }
168        return nodes;
169    }
170
171    private static IdentityHashMap<SpanNode, Integer> getSpanNodes(SpanTree otherToCopy) {
172        IdentityHashMap<SpanNode, Integer> map = new IdentityHashMap<SpanNode, Integer>();
173        int spanNodeCounter = 0;
174        map.put(otherToCopy.getRoot(), spanNodeCounter++);
175        Iterator<SpanNode> it = otherToCopy.getRoot().childIteratorRecursive();
176        while (it.hasNext()) {
177            map.put(it.next(), spanNodeCounter++);
178        }
179        return map;
180    }

182    private SpanNode copySpan(SpanNode spanTree) {
183        if (spanTree instanceof Span) {
184            return new Span((Span) spanTree);
185        } else if (spanTree instanceof AlternateSpanList) {
186            return new AlternateSpanList((AlternateSpanList) spanTree);
187        } else if (spanTree instanceof SpanList) {
188            return new SpanList((SpanList) spanTree);
189        } else if (spanTree instanceof DummySpanNode) {
190            return spanTree;  //shouldn't really happen
191        } else {
192            throw new IllegalStateException("Cannot create copy of " + spanTree + " with class "
193            + ((spanTree == null) ? "null" : spanTree.getClass()));
194        }
195    }

211    public void setRoot(SpanNode root) {
212        if (!root.isValid()) {
213            throw new IllegalStateException("Cannot use invalid node " + root + " as root node.");
214        }
215        if (root.getParent() != null) {
216            if (root.getParent() != this) {
217                throw new IllegalStateException(root + " is already a child of " + root.getParent() + ", cannot be root of " + this);
218            }
219        }
220        this.root = root;
221        root.setParent(this);
222    }

248    /**
249     * Ensures consistency of the tree in case SpanNodes have been removed, and there are still
250     * Annotations pointing to them. This method has a maximum upper bound of O(3nm), where n is the
251     * total number of Annotations, and m is the number of SpanNodes that had been removed from the tree.
252     * The lower bound is Omega(n), if no SpanNodes had been removed from the tree.
253     */
254    @SuppressWarnings("unchecked")
255    public void cleanup() {
256        Map<Annotation, Annotation> removedAnnotations = removeAnnotationsThatPointToInvalidSpanNodes();
257
258        //here:
259        //iterate through all annotations;
260        //if any of those have ONLY an annotationreference as its value,
261        //    - null reference
262        //    - remove value from annotation
263        //    - remove annotation and add it to removedAnnotations map
264        if (!removedAnnotations.isEmpty()) {
265            Iterator<Annotation> annotationIt = iterator();
266            while (annotationIt.hasNext()) {
267                Annotation a = annotationIt.next();
268                if (!a.hasFieldValue()) {
269                    continue;
270                }
271                FieldValue value = a.getFieldValue();
272
273                if (value instanceof AnnotationReference) {
274                    //the annotation "a" has a reference
275                    AnnotationReference ref = (AnnotationReference) value;
276                    if (removedAnnotations.get(ref.getReference()) != null) {
277                        //this reference refers to a dead annotation
278                        ref.setReference(null);
279                        a.setFieldValue(null);
280                        if (!a.isSpanNodeValid()) {
281                            //this annotation has no span node, delete it
282                            annotationIt.remove();
283                            removedAnnotations.put(a, a);
284                        }
285                    }
286                }
287            }
288        }
289
290        //there may still be references to removed annotations,
291        //inside maps, weighted sets, structs, etc.
292        //if any of those have such references,
293        //   - null reference
294        //   - remove annotationref from struct, map, etc.
295        //   - apart from this, keep struct, map etc. and annotation
296        if (!removedAnnotations.isEmpty()) {
297            for (Annotation a : this) {
298                if (!a.hasFieldValue()) {
299                    continue;
300                }
301                removeObsoleteReferencesFromFieldValue(a.getFieldValue(), removedAnnotations, true);
302            }
303        }
304        //was any annotations removed from the global list? do we still have references to those annotations
305        //that have been removed? if so, remove the references
306        removeAnnotationReferencesThatPointToRemovedAnnotations();
307    }

309    private boolean removeObsoleteReferencesFromFieldValue(FieldValue value, Map<Annotation, Annotation> selectedAnnotations, boolean removeIfPresent) {
310        if (value == null) {
311            return false;
312        }
313
314        if (value.getDataType() instanceof AnnotationReferenceDataType) {
315            AnnotationReference ref = (AnnotationReference) value;
316            if (removeIfPresent) {
317                if (selectedAnnotations.containsValue(ref.getReference())) {
318                    //this reference refers to a dead annotation
319                    ref.setReference(null);
320                    return true;
321                }
322            } else {
323                if (!selectedAnnotations.containsValue(ref.getReference())) {
324                    //this reference refers to a dead annotation
325                    ref.setReference(null);
326                    return true;
327                }
328            }
329        } else if (value.getDataType() instanceof StructuredDataType) {
330            removeObsoleteReferencesFromStructuredType((StructuredFieldValue) value, selectedAnnotations, removeIfPresent);
331        } else if (value.getDataType() instanceof CollectionDataType) {
332            removeObsoleteReferencesFromCollectionType((CollectionFieldValue) value, selectedAnnotations, removeIfPresent);
333        } else if (value.getDataType() instanceof MapDataType) {
334            removeObsoleteReferencesFromMapType((MapFieldValue) value, selectedAnnotations, removeIfPresent);
335        }
336        return false;
337    }
338
339    private boolean removeObsoleteReferencesFromStructuredType(StructuredFieldValue struct, Map<Annotation, Annotation> selectedAnnotations, boolean removeIfPresent) {
340        for (Field f : struct.getDataType().getFields()) {
341            FieldValue fValue = struct.getFieldValue(f);
342            if (removeObsoleteReferencesFromFieldValue(fValue, selectedAnnotations, removeIfPresent)) {
343                struct.removeFieldValue(f);
344            }
345        }
346        return false;
347    }
348
349    private boolean removeObsoleteReferencesFromCollectionType(CollectionFieldValue collection, Map<Annotation, Annotation> selectedAnnotations, boolean removeIfPresent) {
350        Iterator it = collection.fieldValueIterator();
351        while (it.hasNext()) {
352            FieldValue fValue = (FieldValue) it.next();
353            if (removeObsoleteReferencesFromFieldValue(fValue, selectedAnnotations, removeIfPresent)) {
354                it.remove();
355            }
356        }
357        return false;
358    }
359
360    private boolean removeObsoleteReferencesFromMapType(MapFieldValue map, Map<Annotation, Annotation> selectedAnnotations, boolean removeIfPresent) {
361        Iterator valueIt = map.values().iterator();
362        while (valueIt.hasNext()) {
363            FieldValue fValue = (FieldValue) valueIt.next();
364            if (removeObsoleteReferencesFromFieldValue(fValue, selectedAnnotations, removeIfPresent)) {
365                valueIt.remove();
366            }
367        }
368        return false;
369    }

371    @SuppressWarnings("unchecked")
372    private Map<Annotation, Annotation> removeAnnotationsThatPointToInvalidSpanNodes() {
373        Map<Annotation, Annotation> removedAnnotations = new IdentityHashMap<Annotation, Annotation>();
374
375        Iterator<Annotation> annotationIt = iterator();
376        while (annotationIt.hasNext()) {
377            Annotation a = annotationIt.next();
378            if (a.hasSpanNode() && !a.isSpanNodeValid()) {
379                a.setSpanNode(null);
380                a.setFieldValue(null);
381                removedAnnotations.put(a, a);
382                annotationIt.remove();
383            }
384        }
385        return removedAnnotations;
386    }

388    private boolean hasAnyFieldValues() {
389        for (Annotation a : this) {
390            if (a.hasFieldValue()) {
391                return true;
392            }
393        }
394        return false;
395    }
396    private void removeAnnotationReferencesThatPointToRemovedAnnotations() {
397        if (hasAnyFieldValues()) {
398            Map<Annotation, Annotation> annotationsStillPresent = new IdentityHashMap<Annotation, Annotation>();
399            for (Annotation a : this) {
400                annotationsStillPresent.put(a, a);
401            }
402            for (Annotation a : this) {
403                if (!a.hasFieldValue()) {
404                    continue;
405                }
406                //do we have any references to annotations that are NOT in this global list??
407                removeObsoleteReferencesFromFieldValue(a.getFieldValue(), annotationsStillPresent, false);
408            }
409        }
410    }

610    public void createIndex(IndexKey key) {
611        if (key == IndexKey.SPAN_NODE && annotations instanceof ListAnnotationContainer) {
612            AnnotationContainer tmpAnnotations = new SpanNode2AnnotationContainer();
613            tmpAnnotations.annotateAll(annotations.annotations());
614            annotations = tmpAnnotations;
615        } else if (key == IndexKey.ANNOTATION_TYPE && annotations instanceof ListAnnotationContainer) {
616            AnnotationContainer tmpAnnotations = new AnnotationType2AnnotationContainer();
617            tmpAnnotations.annotateAll(annotations.annotations());
618            annotations = tmpAnnotations;
619        } else {
620            throw new IllegalArgumentException("Multiple indexes not yet supported. Use clearIndex() or clearIndexes() first.");
621        }
622    }
623
624    public void clearIndex(IndexKey key) {
625        if (key == IndexKey.SPAN_NODE && annotations instanceof SpanNode2AnnotationContainer) {
626            clearIndex();
627        } else if (key == IndexKey.ANNOTATION_TYPE && annotations instanceof AnnotationType2AnnotationContainer) {
628            clearIndex();
629        }
630    }

632    public void clearIndexes() {
633        if (!(annotations instanceof ListAnnotationContainer)) {
634            clearIndex();
635        }
636    }

638    private void clearIndex() {
639        AnnotationContainer tmpAnnotations = new ListAnnotationContainer();
640        tmpAnnotations.annotateAll(annotations.annotations());
641        annotations = tmpAnnotations;
642    }

644    public Collection<IndexKey> getCurrentIndexes() {
645        if (annotations instanceof AnnotationType2AnnotationContainer)
646            return ImmutableList.of(IndexKey.ANNOTATION_TYPE);
647        if (annotations instanceof SpanNode2AnnotationContainer)
648            return ImmutableList.of(IndexKey.SPAN_NODE);
649        return ImmutableList.of();
650    }

657    @Override
658    public boolean equals(Object o) {
659        if (this == o) return true;
660        if (!(o instanceof SpanTree)) return false;

662        SpanTree tree = (SpanTree) o;
663        if (!annotationsEquals(tree)) return false;
664        if (!name.equals(tree.name)) return false;
665        if (!root.equals(tree.root)) return false;

667        return true;

670    @SuppressWarnings("unchecked")
671    private boolean annotationsEquals(SpanTree tree) {
672        List<Annotation> annotationCollection = new LinkedList<Annotation>(getAnnotations());
673        List<Annotation> otherAnnotations = new LinkedList<Annotation>(tree.getAnnotations());
674
675        return annotationCollection.size() == otherAnnotations.size() && CollectionUtils.isEqualCollection(annotationCollection, otherAnnotations);
676    }

678    @Override
679    public int hashCode() {
680        int result = name.hashCode();
681        result = 31 * result + root.hashCode();
682        result = 31 * result + annotations.hashCode();
683        return result;
684    }

686    @Override
687    public int compareTo(SpanTree spanTree) {
688        int comp = name.compareTo(spanTree.name);
689        if (comp != 0) {
690            comp = root.compareTo(spanTree.root);
691        }
692        return comp;
693    }
