002package com.yahoo.document;
003
004import com.yahoo.text.Lowercase;
005import com.yahoo.text.Utf8Array;
006import com.yahoo.text.Utf8String;
007
008import java.io.Serializable;
009
010/**
011 * A full document type name, consisting of a <i>name</i> and a <i>version</i>. The name is case insensitive, and the
012 * version must be a positive integer or 0. This is a <i>value object</i>.
013 *
014 * @author bratseth
015 */
016public final class DataTypeName implements Serializable {
017
018    private final Utf8String name;
019
020    /**
021     * Creates a document name from a string of the form "name"
022     *
023     * @param name The name string to parse.
024     * @throws NumberFormatException if the version part of the name is present but is not a number
025     */
026    public DataTypeName(String name) {
027        this.name = new Utf8String(name);
028    }
029    public DataTypeName(Utf8Array name) {
030        this.name = new Utf8String(name);
031    }
032    public DataTypeName(Utf8String name) {
033        this.name = new Utf8String(name);
034    }
035
036    public String getName() { return name.toString(); }
037
038    @Override
039    public String toString() { return name.toString(); }
040
041    @Override
042    public int hashCode() { return name.hashCode(); }
043
044    @Override
045    public boolean equals(Object obj) {
046        if (!(obj instanceof DataTypeName)) return false;
047        DataTypeName datatype = (DataTypeName)obj;
048        return this.name.equals(datatype.name);
049    }
050
051}
