002package com.yahoo.document;
003
004import com.yahoo.compress.CompressionType;
005import com.yahoo.compress.Compressor;
006
007import java.util.Collection;
008import java.util.LinkedHashMap;
009import java.util.Map;
010
011/**
012 * Date: Apr 15, 2008
013 *
014 * @author humbe
015 */
016public abstract class BaseStructDataType extends StructuredDataType {
017
018    protected Map<Integer, Field> fieldIds = new LinkedHashMap<>();
019    protected Map<Integer, Field> fieldIdsV6 = new LinkedHashMap<>();
020    protected Map<String, Field> fields = new LinkedHashMap<>();
021
022    protected Compressor compressor = new Compressor(CompressionType.NONE);
023
024    BaseStructDataType(String name) {
025        super(name);
026    }
027
028    BaseStructDataType(int id, String name) {
029        super(id, name);
030    }
031
032    protected void assign(BaseStructDataType type) {
033        BaseStructDataType stype = type.clone();
034
035        fieldIds = stype.fieldIds;
036        fieldIdsV6 = stype.fieldIdsV6;
037        fields = stype.fields;
038    }
039
040    @Override
041    public BaseStructDataType clone() {
042        BaseStructDataType type = (BaseStructDataType) super.clone();
043        type.fieldIds = new LinkedHashMap<>();
044        type.fieldIdsV6 = new LinkedHashMap<>();
045
046        type.fields = new LinkedHashMap<>();
047        for (Field field : fieldIds.values()) {
048            type.fields.put(field.getName(), field);
049            type.fieldIds.put(field.getId(Document.SERIALIZED_VERSION), field);
050            type.fieldIdsV6.put(field.getId(6), field);
051        }
052        return type;
053    }
054
055    public void addField(Field field) {
056        if (fields.containsKey(field.getName())) {
057            throw new IllegalArgumentException("Struct " + getName() + " already contains field with name " + field.getName());
058        }
059        if (fieldIds.containsKey(field.getId(Document.SERIALIZED_VERSION))) {
060            throw new IllegalArgumentException("Struct " + getName() + " already contains field with id " + field.getId(Document.SERIALIZED_VERSION));
061        }
062        if (fieldIdsV6.containsKey(field.getId(6))) {
063            throw new IllegalArgumentException("Struct " + getName() + " already contains a field with deprecated document serialization id " + field.getId(6));
064        }
065
066        fields.put(field.getName(), field);
067        fieldIds.put(field.getId(Document.SERIALIZED_VERSION), field);
068        fieldIdsV6.put(field.getId(6), field);
069    }
070
071    public Field removeField(String fieldName) {
072        Field old = fields.remove(fieldName);
073        if (old != null) {
074            fieldIds.remove(old.getId(Document.SERIALIZED_VERSION));
075            fieldIdsV6.remove(old.getId(6));
076        }
077        return old;
078    }
079
080    public void clearFields() {
081        fieldIds.clear();
082        fieldIdsV6.clear();
083        fields.clear();
084    }
085
086    public Field getField(Integer fieldId, int version) {
087        if (version > 6) {
088            return fieldIds.get(fieldId);
089        } else {
090            return fieldIdsV6.get(fieldId);
091        }
092    }
093
094    @Override
095    public Field getField(String fieldName) {
096        return fields.get(fieldName);
097    }
098
099    @Override
100    public Field getField(int id) {
101        return fieldIds.get(id);
102    }
103
104    public boolean hasField(Field field, int version) {
105        Field f = getField(field.getId(version), version);
106        return f != null && f.equals(field);
107    }
108
109    public boolean hasField(String name) {
110        return fields.containsKey(name);
111    }
112
113    public boolean hasField(Field f) {
114        if (hasField(f, 6) || hasField(f, Document.SERIALIZED_VERSION)) {
115            return true;
116        }
117        return false;
118    }
119
120    @Override
121    public Collection<Field> getFields() {
122        return fields.values();
123    }
124
125    public int getFieldCount() {
126        return fields.size();
127    }
128
129    /** Returns the compressor to use to compress data of this type */
130    public Compressor getCompressor() { return compressor; }
131
132    /** Returns a view of the configuration of the compressor used to compress this type */
133    public CompressionConfig getCompressionConfig() {
134        // CompressionConfig accepts a percentage (but exposes a factor) ...
135        float compressionThresholdPercentage = (float)compressor.compressionThresholdFactor() * 100;
136
137        return new CompressionConfig(compressor.type(),
138                                     compressor.level(),
139                                     compressionThresholdPercentage,
140                                     compressor.compressMinSizeBytes());
141    }
142
143    /** Set the config to the compressor used to compress data of this type */
144    public void setCompressionConfig(CompressionConfig config) {
145        CompressionType type = config.type;
146        compressor = new Compressor(type,
147                                    config.compressionLevel,
148                                    config.thresholdFactor(),
149                                    (int)config.minsize);
150    }
151
152}
