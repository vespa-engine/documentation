002package com.yahoo.persistence.spi;
003
004import com.yahoo.document.*;
005import com.yahoo.document.fieldset.AllFields;
006import com.yahoo.persistence.spi.*;
007import com.yahoo.persistence.spi.result.*;
008
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * An abstract class that implements persistence provider functionality that some providers
014 * may not have use for.
015 */
016public abstract class AbstractPersistenceProvider implements PersistenceProvider {
017    @Override
018    public Result initialize() {
019        return new Result();
020    }
021
022    @Override
023    public PartitionStateListResult getPartitionStates() {
024        List<PartitionState> partitionStates = new ArrayList<PartitionState>();
025        partitionStates.add(new PartitionState(PartitionState.State.UP, ""));
026        return new PartitionStateListResult(partitionStates);
027    }
028
029    @Override
030    public Result setClusterState(ClusterState state) {
031        return new Result();
032    }
033
034    @Override
035    public Result setActiveState(Bucket bucket, BucketInfo.ActiveState active) {
036        return new Result();
037    }
038
039
040    @Override
041    public RemoveResult removeIfFound(Bucket bucket, long timestamp, DocumentId id) {
042        return remove(bucket, timestamp, id);
043    }
044
045    @Override
046    public Result removeEntry(Bucket bucket, long timestampToRemove) {
047        return new Result();
048    }
049
050    @Override
051    public Result flush(Bucket bucket) {
052        return new Result();
053    }
054
055    @Override
056    public BucketIdListResult getModifiedBuckets() {
057        return new BucketIdListResult(new ArrayList<BucketId>());
058    }
059
060    @Override
061    public Result maintain(Bucket bucket, MaintenanceLevel level) {
062        return new Result();
063    }
064
065    @Override
066    public Result move(Bucket bucket, short partitionId) {
067        return new Result();
068    }
069
070    @Override
071    public UpdateResult update(Bucket bucket, long timestamp, DocumentUpdate update) {
072        GetResult result = get(bucket, new AllFields(), update.getId());
073        if (result.wasFound()) {
074            Document doc = result.getDocument().clone();
075            update.applyTo(doc);
076            put(bucket, timestamp, doc);
077            return new UpdateResult(result.getLastModifiedTimestamp());
078        } else {
079            return new UpdateResult();
080        }
081    }
082}
