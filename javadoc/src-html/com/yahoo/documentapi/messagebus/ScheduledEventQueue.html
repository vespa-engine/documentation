002package com.yahoo.documentapi.messagebus;
003
004import com.yahoo.concurrent.SystemTimer;
005import com.yahoo.concurrent.Timer;
006
007import java.util.*;
008import java.util.concurrent.RejectedExecutionException;
009
010/**
011 * Simple, lightweight event scheduler that does not maintain any executor
012 * threads of its own, but rather makes it the responsibility of the caller
013 * to run the events as the queue hands them over.
014 *
015 * Fully thread safe for multiple readers and writers.
016 */
017public class ScheduledEventQueue {
018    private final Set<Entry> tasks = new TreeSet<Entry>();
019    private long sequenceCounter = 0;
020    private Timer timer;
021    private volatile boolean waiting = false;
022    private volatile boolean shutdown = false;
023
024    private static class Entry implements Comparable<Entry> {
025        private Runnable task;
026        private long timestamp;
027        private long sequenceId;
028
029        public Entry(Runnable task, long timestamp, long sequenceId) {
030            this.task = task;
031            this.timestamp = timestamp;
032            this.sequenceId = sequenceId;
033        }
034
035        @Override
036        public boolean equals(Object o) {
037            if (this == o) return true;
038            if (o == null || getClass() != o.getClass()) return false;
039
040            Entry entry = (Entry) o;
041
042            if (sequenceId != entry.sequenceId) return false;
043            if (timestamp != entry.timestamp) return false;
044            if (!task.equals(entry.task)) return false;
045
046            return true;
047        }
048
049        public int compareTo(Entry o) {
050            if (timestamp < o.timestamp) return -1;
051            if (timestamp > o.timestamp) return 1;
052            if (sequenceId < o.sequenceId) return -1;
053            if (sequenceId > o.sequenceId) return 1;
054            return 0;
055        }
056
057        public Runnable getTask() {
058            return task;
059        }
060
061        public long getTimestamp() {
062            return timestamp;
063        }
064
065        public long getSequenceId() {
066            return sequenceId;
067        }
068    }
069
070    public ScheduledEventQueue() {
071        this.timer = SystemTimer.INSTANCE;
072    }
073
074    public ScheduledEventQueue(Timer timer) {
075        this.timer = timer;
076    }
077
078    public void pushTask(Runnable task) {
079        synchronized (tasks) {
080            if (shutdown) {
081                throw new RejectedExecutionException("Tasks can't be scheduled since queue has been shut down.");
082            }
083
084            tasks.add(new Entry(task, 0, sequenceCounter++));
085            tasks.notifyAll();
086        }
087    }
088
089    public void pushTask(Runnable task, long milliSecondsToWait) {
090        synchronized (tasks) {
091            if (shutdown) {
092                throw new RejectedExecutionException("Tasks can't be scheduled since queue has been shut down.");
093            }
094
095            tasks.add(new Entry(task, timer.milliTime() + milliSecondsToWait, sequenceCounter++));
096            tasks.notifyAll();
097        }
098    }
099
100    public boolean isWaiting() {
101        synchronized (tasks) {
102            return waiting;
103        }
104    }
105
106    /**
107     * Waits until the queue has a task that is ready for scheduling, removes that
108     * task from the queue and returns it.
109     *
110     * @return The next task.
111     */
112    public Runnable getNextTask() {
113        try {
114            while (true) {
115                synchronized (tasks) {
116                    Iterator<Entry> iter = tasks.iterator();
117                    if (!iter.hasNext()) {
118                        if (shutdown) {
119                            return null;
120                        }
121                        // Set flag for unit tests to coordinate with.
122                        waiting = true;
123                        tasks.wait();
124                        waiting = false;
125                        continue;
126                    }
127                    Entry retEntry = iter.next();
128                    long timeNow = timer.milliTime();
129                    if (retEntry.getTimestamp() > timeNow) {
130                        waiting = true;
131                        tasks.wait(retEntry.getTimestamp() - timeNow);
132                        waiting = false;
133                        continue;
134                    }
135                    iter.remove();
136                    return retEntry.getTask();
137                }
138            }
139        } catch (InterruptedException e) {
140            return null;
141        }
142    }
143
144    /**
145     * If there is a task ready for scheduling, remove it from the queue and return it.
146     *
147     * @return The next task.
148     */
149    public Runnable popTask() {
150        synchronized (tasks) {
151            Iterator<Entry> iter = tasks.iterator();
152            if (!iter.hasNext()) {
153                return null;
154            }
155            Entry retEntry = iter.next();
156            if (retEntry.getTimestamp() > timer.milliTime()) {
157                return null;
158            }
159            iter.remove();
160            return retEntry.getTask();
161        }
162    }
163
164    /** For unit testing only */
165    public void wakeTasks() {
166        synchronized (tasks) {
167            tasks.notifyAll();
168        }
169    }
170
171    public void shutdown() {
172        synchronized (tasks) {
173            shutdown = true;
174            tasks.notifyAll();
175        }
176    }
177
178    public boolean isShutdown() {
179        synchronized (tasks) {
180            return shutdown;
181        }
182    }
183
184    public long size() {
185        synchronized (tasks) {
186            return tasks.size();
187        }
188    }
189}
