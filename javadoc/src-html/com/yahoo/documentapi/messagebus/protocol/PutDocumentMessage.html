002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.document.Document;
005import com.yahoo.document.DocumentPut;
006import com.yahoo.document.TestAndSetCondition;
007import com.yahoo.document.serialization.DocumentDeserializer;
008
009import java.nio.ByteBuffer;
010import java.util.Arrays;
011
012/**
013 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
014 */
015public class PutDocumentMessage extends TestAndSetMessage {
016
017    private DocumentDeserializer buffer = null;
018    private DocumentPut put = null;
019    private long time = 0;
020    private LazyDecoder decoder = null;
021
022    /**
023     * Constructs a new message for deserialization.
024     */
025    PutDocumentMessage() {
026        // empty
027    }
028
029    /**
030     * Constructs a new message from a byte buffer.
031     *  @param decoder The decoder to use for deserialization.
032     * @param buffer  A byte buffer that contains a serialized message.
033     */
034    public PutDocumentMessage(LazyDecoder decoder, DocumentDeserializer buffer) {
035        this.decoder = decoder;
036        this.buffer = buffer;
037    }
038
039    /**
040     * Constructs a new document put message.
041     *
042     * @param put Document put operation
043     */
044    public PutDocumentMessage(DocumentPut put) {
045        this.put = put;
046    }
047
048    /**
049     * Creates an empty PutDocumentMessage
050     */
051    public static PutDocumentMessage createEmpty() {
052        return new PutDocumentMessage(null);
053    }
054
055    /**
056     * This method will make sure that any serialized content is deserialized into proper message content on first
057     * entry. Any subsequent entry into this function will do nothing.
058     */
059    private void deserialize() {
060        if (decoder != null && buffer != null) {
061            decoder.decode(this, buffer);
062            decoder = null;
063            buffer = null;
064        }
065    }
066
067    /**
068     * Returns the document put operation
069     */
070    public DocumentPut getDocumentPut() {
071        deserialize();
072        return put;
073    }
074
075    /**
076     * Sets the document to put.
077     *
078     * @param put Put document operation
079     */
080    public void setDocumentPut(DocumentPut put) {
081        buffer = null;
082        decoder = null;
083        this.put = put;
084    }
085
086    /**
087     * Returns the timestamp of the document to put.
088     *
089     * @return The document timestamp.
090     */
091    public long getTimestamp() {
092        deserialize();
093        return time;
094    }
095
096    /**
097     * Sets the timestamp of the document to put.
098     *
099     * @param time The timestamp to set.
100     */
101    public void setTimestamp(long time) {
102        buffer = null;
103        decoder = null;
104        this.time = time;
105    }
106
107    /**
108     * Returns the raw serialized buffer. This buffer is stored as the message is received from accross the network, and
109     * deserialized from as soon as a member is requested. This method will return null if the buffer has been decoded.
110     *
111     * @return The buffer containing the serialized data for this message, or null.
112     */
113    ByteBuffer getSerializedBuffer() {
114        return buffer != null ? buffer.getBuf().getByteBuffer() : null; // TODO: very dirty. Must make interface.
115    }
116
117    @Override
118    public DocumentReply createReply() {
119        return new WriteDocumentReply(DocumentProtocol.REPLY_PUTDOCUMENT);
120    }
121
122    @Override
123    public int getApproxSize() {
124        if (buffer != null) {
125            return buffer.getBuf().remaining();
126        }
127        return put.getDocument().getApproxSize();
128    }
129
130    @Override
131    public boolean hasSequenceId() {
132        return true;
133    }
134
135    @Override
136    public long getSequenceId() {
137        deserialize();
138        return Arrays.hashCode(put.getId().getGlobalId());
139    }
140
141    @Override
142    public int getType() {
143        return DocumentProtocol.MESSAGE_PUTDOCUMENT;
144    }
145
146    @Override
147    public TestAndSetCondition getCondition() {
148        deserialize();
149        return put.getCondition();
150    }
151
152    @Override
153    public void setCondition(TestAndSetCondition condition) {
154        put.setCondition(condition);
155    }
156}
