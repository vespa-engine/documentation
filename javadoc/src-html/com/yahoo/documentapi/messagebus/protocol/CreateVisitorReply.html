002package com.yahoo.documentapi.messagebus.protocol;
003
004import com.yahoo.document.BucketId;
005import com.yahoo.vdslib.VisitorStatistics;
006
007public class CreateVisitorReply extends DocumentReply {
008
009    private BucketId lastBucket;
010    private VisitorStatistics statistics = new VisitorStatistics();
011
012    public CreateVisitorReply(int type) {
013        super(type);
014        lastBucket = new BucketId(Integer.MAX_VALUE);
015    }
016
017    public void setLastBucket(BucketId lastBucket) {
018        this.lastBucket = lastBucket;
019    }
020
021    public BucketId getLastBucket() {
022        return lastBucket;
023    }
024
025    public void setVisitorStatistics(VisitorStatistics statistics) {
026        this.statistics = statistics;
027    }
028
029    public VisitorStatistics getVisitorStatistics() {
030        return statistics;
031    }
032
033
034}
