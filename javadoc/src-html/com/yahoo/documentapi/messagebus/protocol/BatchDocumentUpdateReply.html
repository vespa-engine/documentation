002package com.yahoo.documentapi.messagebus.protocol;
003
004import java.util.ArrayList;
005
006public class BatchDocumentUpdateReply extends WriteDocumentReply {
007
008    private ArrayList<Boolean> documentsNotFound = new ArrayList<Boolean>();
009
010    /**
011     * Constructs a new reply with no content.
012     */
013    public BatchDocumentUpdateReply() {
014        super(DocumentProtocol.REPLY_BATCHDOCUMENTUPDATE);
015    }
016
017    /**
018     * If all documents to update are found, this vector will be empty. If
019     * one or more documents are not found, this vector will have the size of
020     * the initial number of updates, with entries set to true where the
021     * corresponding update was not found.
022     *
023     * @return Vector containing indices of not found documents, or empty
024     *   if all documents were found
025     */
026    public ArrayList<Boolean> getDocumentsNotFound() {
027        return documentsNotFound;
028    }
029}
