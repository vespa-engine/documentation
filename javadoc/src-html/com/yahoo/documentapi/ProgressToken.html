015/**
016 * Token to use to keep track of progress for visiting. Can be used to resume
017 * visiting if visiting has been aborted for any reason.
018 *
019 * @author <a href="mailto:thomasg@yahoo-inc.com">Thomas Gundersen</a>

024    private static final Logger log = Logger.getLogger(ProgressToken.class.getName());
025    /**
026     * Any bucket kept track of by a <code>ProgressToken</code> instance may
027     * be in one of two states: pending or active. <em>Pending</em> means that
028     * a bucket may be returned by a VisitorIterator, i.e. it is ready for
029     * visiting, while <em>active</em> means that a bucket is currently being
030     * visited and may thus not be returned from an iterator.
031     *
032     * Getting a pending bucket via the iterator sets its state to active and
033     * updating an active bucket sets its state back to pending again.
034     */
035    public enum BucketState {
036        BUCKET_PENDING,
037        BUCKET_ACTIVE
038    }
039
040    public static final BucketId NULL_BUCKET = new BucketId();
041    public static final BucketId FINISHED_BUCKET = new BucketId(Integer.MAX_VALUE);

046     * {@link com.yahoo.documentapi.VisitorIterator#getNext()} but have not
047     * yet been update()'d) and pending buckets (i.e. those that have been
048     * update()'d and may be returned by getNext() at some point)
049     */

076    /**
077     * For consistent bucket key ordering, we need to ensure that reverse bucket
078     * IDs that have their MSB set actually are compared as being greater than
079     * those that don't. This is yet another issue caused by Java's lack of
080     * unsigned integers.
081     */
082    public static class BucketKeyWrapper implements Comparable<BucketKeyWrapper>
083    {
084        private long key;
085
086        public BucketKeyWrapper(long key) {
087            this.key = key;
088        }
089
090        public int compareTo(BucketKeyWrapper other) {
091            if ((key & 0x8000000000000000L) != (other.key & 0x8000000000000000L)) {
092                // MSBs differ
093                return ((key >>> 63) > (other.key >>> 63)) ? 1 : -1;
094            }
095            // Mask off MSBs since we've already checked them, and with MSB != 1
096            // we know the ordering will be consistent
097            if ((key & 0x7FFFFFFFFFFFFFFFL) < (other.key & 0x7FFFFFFFFFFFFFFFL)) {
098                return -1;
099            } else if ((key & 0x7FFFFFFFFFFFFFFFL) > (other.key & 0x7FFFFFFFFFFFFFFFL)) {
100                return 1;
101            }
102            return 0;
103        }
104
105        public long getKey() {
106            return key;
107        }
108
109        public BucketId toBucketId() {
110            return new BucketId(keyToBucketId(key));
111        }
112
113        @Override
114        public String toString() {
115            return Long.toHexString(key);
116        }
117
118        @Override
119        public boolean equals(Object o) {
120            if (this == o) return true;
121            if (o == null || !(o instanceof BucketKeyWrapper)) return false;
122            return key == ((BucketKeyWrapper)o).key;
123        }
124
125        @Override
126        public int hashCode() {
127            return (int) (key ^ (key >>> 32));
128        }
129    }

168    public ProgressToken(String serialized) {
169        String[] lines = serialized.split("\\n");
170        if (lines.length < 5) {
171            throw new IllegalArgumentException("Progress file is malformed or a deprecated version");
172        }
173
174        // 1st token is simple header text. Just check that it starts with
175        // a known value. To be 5.0 backwards compatible, we do not check
176        // the rest of the line.
177        final String header = lines[0];
178        if (!header.startsWith("VDS bucket progress file")) {
179            throw new IllegalArgumentException("File does not appear to be a " +
180                    "valid VDS progress file; expected first line to start with " +
181                    "'VDS bucket progress file'");
182        }
183        // 2nd token contains the distribution bit count the progress file was
184        // saved with
185        distributionBits = Integer.parseInt(lines[1]);
186        bucketCursor = Long.parseLong(lines[2]);
187        finishedBucketCount = Long.parseLong(lines[3]);
188        totalBucketCount = Long.parseLong(lines[4]);
189
190        if (totalBucketCount == finishedBucketCount) {
191            return; // We're done here
192        }
193
194        // The rest of the tokens are super:sub bucket progress pairs
195        for (int i = 5; i < lines.length; ++i) {
196            String[] buckets = lines[i].split(":");
197            if (buckets.length != 2) {
198                throw new IllegalArgumentException("Bucket progress file contained malformed line");
199            }
200            // Due to Java's fantastically broken handling of unsigned integer
201            // conversion, the following workaround (i.e. hack) is used for now
202            // (it was also used in the past for presumably the same reason).
203            BucketId superId = new BucketId("BucketId(0x" + buckets[0] + ")");
204            BucketId subId;
205            if ("0".equals(buckets[1])) {
206                subId = new BucketId();
207            } else {
208                subId = new BucketId("BucketId(0x" + buckets[1] + ")");
209            }
210            addBucket(superId, subId, BucketState.BUCKET_PENDING);
211        }
212    }

214    public ProgressToken(byte[] serialized) {
215        DocumentDeserializer in = DocumentDeserializerFactory.create42(null, GrowableByteBuffer.wrap(serialized));
216        distributionBits = in.getInt(null);
217        bucketCursor = in.getLong(null);
218        finishedBucketCount = in.getLong(null);
219        totalBucketCount = in.getLong(null);
220
221        int progressCount = in.getInt(null);
222        for (int i = 0; i < progressCount; ++i) {
223            long key = in.getLong(null);
224            long value = in.getLong(null);
225            addBucket(new BucketId(key), new BucketId(value), BucketState.BUCKET_PENDING);
226        }
227    }
228
229    public byte[] serialize() {
230        DocumentSerializer out = DocumentSerializerFactory.create42(new GrowableByteBuffer());
231        out.putInt(null, distributionBits);
232        out.putLong(null, bucketCursor);
233        out.putLong(null, finishedBucketCount);
234        out.putLong(null, totalBucketCount);
235
236        out.putInt(null, buckets.size());
237
238        // Append individual bucket progress
239        for (Map.Entry<BucketKeyWrapper, ProgressToken.BucketEntry> entry : buckets.entrySet()) {
240            out.putLong(null, keyToBucketId(entry.getKey().getKey()));
241            out.putLong(null, entry.getValue().getProgress().getRawId());
242        }
243
244        byte[] ret = new byte[out.getBuf().position()];
245        out.getBuf().rewind();
246        out.getBuf().get(ret);
247        return ret;
248    }

250    public void addFailedBucket(BucketId superbucket, BucketId progress, String errorMsg) {
251        BucketId existing = failedBuckets.put(superbucket, progress);
252        if (existing != null) {
253            throw new IllegalStateException(
254                    "Attempting to add a superbucket to failed buckets that has already been added: "
255                            + superbucket + ":" + progress);
256        }
257        if (firstErrorMsg == null) {
258            firstErrorMsg = errorMsg;
259        }
260    }

270    /**
271     * Updates internal progress state for <code>bucket</code>, indicating it's currently
272     * at <code>progress</code>. Assumes that given a completely finished bucket, this
273     * function will not be called again to further update its progress after
274     * the finished-update.
275     *
276     * @see VisitorIterator#update(com.yahoo.document.BucketId, com.yahoo.document.BucketId)
277     *
278     * @param superbucket A valid superbucket ID that exists in <code>buckets</code>
279     * @param progress The sub-bucket progress that has been reached in the
280     * superbucket
281     */
282    protected void updateProgress(BucketId superbucket, BucketId progress) {
283        // There exists a valid case in which the progress bucket may actually contains
284        // its superbucket from the POV of the storage code, so it has to be handled
285        // appropriately.
286        if (!progress.equals(NULL_BUCKET)
287                && !progress.equals(FINISHED_BUCKET)
288                && !superbucket.contains(progress)
289                && !progress.contains(superbucket)) {
290            if (log.isLoggable(LogLevel.DEBUG)) {
291                    log.log(LogLevel.DEBUG, "updateProgress called with non-contained bucket "
292                       + "pair " + superbucket + ":" + progress + ", but allowing anyway");
293                }
294        }
295
296        BucketKeyWrapper superKey = bucketToKeyWrapper(superbucket);
297        BucketEntry entry = buckets.get(superKey);
298        if (entry == null) {
299            throw new IllegalArgumentException(
300                    "updateProgress with unknown superbucket "
301                    + superbucket + ":" + progress);
302        }
303
304        // If progress == Integer.MAX_VALUE, we're done. Otherwise, we're not
305        if (!progress.equals(FINISHED_BUCKET)) {
306            if (entry.getState() != BucketState.BUCKET_ACTIVE) {
307                if (log.isLoggable(LogLevel.DEBUG)) {
308                    log.log(LogLevel.DEBUG, "updateProgress called with sub-bucket that was "
309                       + "not marked as active " + superbucket + ":" + progress);
310                }
311            } else {
312                assert(activeBucketCount > 0);
313                --activeBucketCount;
314                ++pendingBucketCount;
315            }
316            // Mark bucket as pending instead of active, allowing it to be
317            // reused by the iterator
318            entry.setState(BucketState.BUCKET_PENDING);
319            entry.setProgress(progress);
320        }
321        else {
322            // Superbucket is finished, alongside its sub-bucket tree
323            ++finishedBucketCount;
324            if (entry.getState() == BucketState.BUCKET_PENDING) {
325                assert(pendingBucketCount > 0);
326                --pendingBucketCount;
327            } else {
328                assert(activeBucketCount > 0);
329                --activeBucketCount;
330            }
331            buckets.remove(superKey);
332        }
333    }

335    /**
336     * <em>For use internally by DocumentAPI code only</em>. Using this method by
337     * itself will invariably lead to undefined ProgressToken state unless
338     * care is taken. Leave it to the VisitorIterator.
339     *
340     * @param superbucket Superbucket that will be progress-tracked
341     * @param progress Bucket progress thus far
342     * @param state Initial bucket state. Only pending buckets may be returned
343     */
344    protected void addBucket(BucketId superbucket, BucketId progress, BucketState state) {
345        if (progress.equals(FINISHED_BUCKET)) {
346            if (log.isLoggable(LogLevel.DEBUG)) {
347                log.log(LogLevel.DEBUG, "Trying to add already finished superbucket "
348                        + superbucket + "; ignoring it");
349            }
350            return;
351        }
352        if (log.isLoggable(LogLevel.SPAM)) {
353            log.log(LogLevel.SPAM, "Adding bucket pair " + superbucket
354                    + ":" + progress + " with state " + state);
355        }
356
357        BucketEntry entry = new BucketEntry(progress, state);
358        BucketEntry existing = buckets.put(bucketToKeyWrapper(superbucket), entry);
359        if (existing != null) {
360            throw new IllegalStateException(
361                    "Attempting to add a superbucket that has already been added: "
362                    + superbucket + ":" + progress);
363        }
364        if (state == BucketState.BUCKET_PENDING) {
365            ++pendingBucketCount;
366        } else {
367            ++activeBucketCount;
368        }
369    }

372     * Directly generate a bucket Id key for the <code>n</code>th bucket in
373     * reverse sorted order.
374     *
375     * @param n a number in the range [0, 2**<code>distributionBits</code>)
376     * @param distributionBits Distribution bit count for the generated key
377     * @return A value where, if you had generated 2**<code>distributionBits</code>
378     * {@link BucketId}s with incremental numerical IDs and then sorted
379     * them on their reverse bit-order keys, the returned key would be equal
380     * to the nth element in the resulting sorted sequence.
381     */
382    public static long makeNthBucketKey(long n, int distributionBits) {
383        return (n << (64 - distributionBits)) | distributionBits;
384    }

495    /**
496     * The format of the bucket progress output is as follows:
497     * <pre>
498     *   VDS bucket progress file (n% completed)\n
499     *   distribution bit count\n
500     *   current bucket cursor\n
501     *   number of finished buckets\n
502     *   total number of buckets\n
503     *   hex-of-superbucket:hex-of-progress\n
504     *   ... repeat above line for each pending bucket ...
505     * </pre>
506     *
507     * Note that unlike earlier versions of ProgressToken, the bucket IDs are
508     * not prefixed with '0x'.
509     */
510    public synchronized String toString() {
511        StringBuilder sb = new StringBuilder();
512        // Append header
513        sb.append("VDS bucket progress file (");
514        sb.append(percentFinished());
515        sb.append("% completed)\n");
516        sb.append(distributionBits);
517        sb.append('\n');
518        sb.append(bucketCursor);
519        sb.append('\n');
520        long doneBucketCount = Math.max(0l, finishedBucketCount - failedBuckets.size());
521        sb.append(doneBucketCount);
522        sb.append('\n');
523        sb.append(totalBucketCount);
524        sb.append('\n');
525        // Append individual bucket progress
526        for (Map.Entry<BucketKeyWrapper, ProgressToken.BucketEntry> entry : buckets.entrySet()) {
527            sb.append(Long.toHexString(keyToBucketId(entry.getKey().getKey())));
528            sb.append(':');
529            sb.append(Long.toHexString(entry.getValue().getProgress().getRawId()));
530            sb.append('\n');
531        }
532        for (Map.Entry<BucketId, BucketId> entry : failedBuckets.entrySet()) {
533            sb.append(Long.toHexString(entry.getKey().getRawId()));
534            sb.append(':');
535            sb.append(Long.toHexString(entry.getValue().getRawId()));
536            sb.append('\n');
537        }
538        return sb.toString();
539    }

541    /**
542     * Calculate an estimate on how far we've managed to iterate over both the
543     * superbuckets and the sub-buckets.
544     *
545     * Runs in <em>O(n+m)</em> time, where <em>n</em> is the number of active buckets
546     * and <em>m</em> is the number of pending buckets. Both these values should
547     * be fairly small in practice, however.
548     *
549     * Method is synchronized, as legacy code treats this as an atomic read.
550     *
551     * @return A value in the range [0, 100] estimating the progress.
552     */
553    public synchronized double percentFinished() {
554        long superTotal = totalBucketCount;
555        long superFinished = finishedBucketCount;
556
557        if (superTotal == 0 || superTotal == superFinished) return 100;
558
559        double superDelta = 100.0 / superTotal;
560        double cumulativeSubProgress = 0;
561
562        // Calculate cumulative for all non-finished buckets. 0 means the
563        // bucket has yet to see any progress
564        // There are numerical precision issues here, but this hardly requires
565        // aerospace engineering result-accuracy
566        for (Map.Entry<BucketKeyWrapper, ProgressToken.BucketEntry> entry : buckets.entrySet()) {
567            BucketId superbucket = new BucketId(keyToBucketId(entry.getKey().getKey()));
568            BucketId progress = entry.getValue().getProgress();
569            // Prevent calculation of bucket progress on inconsistent buckets
570            if (progress.getId() != 0 && superbucket.contains(progress)) {
571                cumulativeSubProgress += superDelta * progressFraction(superbucket, progress);
572            }
573        }
574
575        return (((double)superFinished / (double)superTotal) * 100.0)
576                + cumulativeSubProgress;
577    }

579    /*
580     * Based on the following C++ code from document/bucket/bucketid.cpp:
581     *
582     * BucketId::Type
583     * BucketId::bucketIdToKey(Type id)
584     * {
585     *   Type retVal = reverse(id);
586     *
587     *   Type usedCountLSB = id >> maxNumBits();
588     *   retVal >>= CountBits;
589     *   retVal <<= CountBits;
590     *   retVal |= usedCountLSB;
591     *
592     *   return retVal;
593     * }
594     *
595     * static uint32_t maxNumBits() { return (8 * sizeof(Type) - CountBits);}
596     */
597    // TODO: this should probably be moved to BucketId at some point?
598    public static long bucketToKey(long id) {
599        long retVal = Long.reverse(id);
600        long usedCountLSB = id >>> (64 - BucketId.COUNT_BITS);
601        retVal >>>= BucketId.COUNT_BITS;
602        retVal <<= BucketId.COUNT_BITS;
603        retVal |= usedCountLSB;
604
605        return retVal;
606    }
607
608    private static BucketKeyWrapper bucketToKeyWrapper(BucketId bucket) {
609        return new BucketKeyWrapper(bucketToKey(bucket.getId()));
610    }
611    /*
612     * BucketId::Type
613     * BucketId::keyToBucketId(Type key)
614     * {
615     *   Type retVal = reverse(key);
616     *
617     *   Type usedCountMSB = key << maxNumBits();
618     *   retVal <<= CountBits;
619     *   retVal >>= CountBits;
620     *   retVal |= usedCountMSB;
621     *
622     *   return retVal;
623     * }
624     */
625    public static long keyToBucketId(long key) {
626        long retVal = Long.reverse(key);
627        long usedCountMSB = key << (64 - BucketId.COUNT_BITS);
628        retVal <<= BucketId.COUNT_BITS;
629        retVal >>>= BucketId.COUNT_BITS;
630        retVal |= usedCountMSB;
631
632        return retVal;
633    }

635    /**
636     * @param superbucket The superbucket of which <code>progress</code> is
637     * a sub-bucket
638     * @param progress The sub-bucket for which a fractional progress should
639     * be calculated
640     * @return a value in [0, 1] specifying how far the (sub-bucket) has
641     * reached in its superbucket. This is calculated by looking at the
642     * bucket's split factor.
643     */
644    public synchronized double progressFraction(BucketId superbucket, BucketId progress) {
645        long revBits = bucketToKey(progress.getId());
646        int superUsed = superbucket.getUsedBits();
647        int progressUsed = progress.getUsedBits();
648
649        if (progressUsed == 0 || progressUsed < superUsed) {
650            return 0;
651        }
652
653        int splitCount = progressUsed - superUsed;
654
655        if (splitCount == 0) return 1; // Superbucket or inconsistent used-bits
656
657        // Extract reversed split-bits
658        revBits <<= superUsed;
659        revBits >>>= 64 - splitCount;
660
661        return (double)(revBits + 1) / (double)(1L << splitCount);
662    }

664    /**
665     * Checks whether or not a given bucket is certain to be finished. Only
666     * looks at the super-bucket part of the given bucket ID, so it's possible
667     * that the bucket has in fact finished on a sub-bucket progress level.
668     * This does not affect the correctness of the result, however.
669     *
670     * During a distribution bit change, the token's buckets may be inconsistent.
671     * In this scenario, false is always returned since we can't tell for
672     * sure if the bucket is still active until the buckets have been made
673     * consistent.
674     *
675     * <strong>FIXME: verify correctness with regards to orderdoc et al.
676     * Don't make this method public until this has been done!</strong>
677     *
678     * @param bucket Bucket to check whether or not is finished.
679     * @return <code>true</code> if <code>bucket</code>'s super-bucket is
680     * finished, <code>false</code> otherwise.
681     */
682    protected synchronized boolean isBucketFinished(BucketId bucket) {
683        if (inconsistentState) {
684            return false;
685        }
686        // Token only knows of super-buckets, not sub buckets
687        BucketId superbucket = new BucketId(distributionBits, bucket.getId());
688        // Bucket is done if the current cursor location implies a visitor for
689        // the associated superbucket has already been sent off at some point
690        // and there is no pending visitor for the superbucket. The cursor is
691        // used to directly generate bucket keys, so we can compare against it
692        // directly.
693        // Example: given db=3 and cursor=2, the buckets 000 and 100 will have
694        // been returned by the iterator. By reversing the id and "right-
695        // aligning" it, we get the cursor location that would be required to
696        // generate it.
697        // We also return false if we're inconsistent, since the active/pending
698        // check is done on exact key values, requiring a uniform distribution
699        // bit value.
700        long reverseId = Long.reverse(superbucket.getId())
701                >>> (64 - distributionBits); // No count bits
702
703        if (reverseId >= bucketCursor) {
704            return false;
705        }
706        // Bucket has been generated, and it must have been finished if it's
707        // not listed as active/pending since we always remove finished buckets
708        BucketEntry entry = buckets.get(bucketToKeyWrapper(superbucket));
709        if (entry == null) {
710            return true;
711        }
712        // If key of bucket progress > key of bucket id, we've finished it
713        long bucketKey = bucketToKey(bucket.getId());
714        long progressKey = bucketToKey(entry.getProgress().getId());
715        // TODO: verify correctness for all bucket orderings!
716        return progressKey > bucketKey;
717    }
718
719    /**
720     *
721     * @param bucket BucketId to be split into two buckets. Bucket's used-bits
722     * do not need to match the ProgressToken's current distribution bit count,
723     * as it is assumed the client knows what it's doing and will bring the
724     * token into a consistent state eventually.
725     */
726    protected void splitPendingBucket(BucketId bucket) {
727        BucketKeyWrapper bucketKey = bucketToKeyWrapper(bucket);
728        BucketEntry entry = buckets.get(bucketKey);
729        if (entry == null) {
730            throw new IllegalArgumentException(
731                    "Attempting to split unknown bucket: " + bucket);
732        }
733        if (entry.getState() != BucketState.BUCKET_PENDING) {
734            throw new IllegalArgumentException(
735                    "Attempting to split non-pending bucket: " + bucket);
736        }
737
738        int splitDistBits = bucket.getUsedBits() + 1;
739        // Original bucket is replaced by two split children
740        BucketId splitLeft = new BucketId(splitDistBits, bucket.getId());
741        // Right split sibling becomes logically at location original_bucket*2 in the
742        // bucket space due to the key ordering and setting the MSB of the split
743        BucketId splitRight = new BucketId(splitDistBits, bucket.getId()
744                | (1L << bucket.getUsedBits()));
745
746        addBucket(splitLeft, entry.getProgress(), BucketState.BUCKET_PENDING);
747        addBucket(splitRight, entry.getProgress(), BucketState.BUCKET_PENDING);
748
749        // Remove old bucket
750        buckets.remove(bucketKey);
751        --pendingBucketCount;
752    }
753
754    protected void mergePendingBucket(BucketId bucket) {
755        BucketKeyWrapper bucketKey = bucketToKeyWrapper(bucket);
756        BucketEntry entry = buckets.get(bucketKey);
757        if (entry == null) {
758            throw new IllegalArgumentException(
759                    "Attempting to join unknown bucket: " + bucket);
760        }
761        if (entry.getState() != BucketState.BUCKET_PENDING) {
762            throw new IllegalArgumentException(
763                    "Attempting to join non-pending bucket: " + bucket);
764        }
765
766        int usedBits = bucket.getUsedBits();
767        // If MSB is 0, we should look for the bucket's right sibling. If not,
768        // we know that there's no left sibling, as it should otherwise have been
769        // merged already by the caller, due to it being ordered before the
770        // right sibling in the pending mapping
771        if ((bucket.getId() & (1L << (usedBits - 1))) == 0) {
772            BucketId rightCheck = new BucketId(usedBits, bucket.getId() | (1L << (usedBits - 1)));
773            BucketEntry rightSibling = buckets.get(bucketToKeyWrapper(rightCheck));
774            // Must not merge if sibling isn't pending
775            if (rightSibling != null) {
776                assert(rightSibling.getState() == BucketState.BUCKET_PENDING);
777                if (log.isLoggable(LogLevel.SPAM)) {
778                    log.log(LogLevel.SPAM, "Merging " + bucket + " with rhs " + rightCheck);
779                }
780                // If right sibling has progress, it will unfortunately have to
781                // be discarded
782                if (rightSibling.getProgress().getUsedBits() != 0
783                        && log.isLoggable(LogLevel.DEBUG)) {
784                    log.log(LogLevel.DEBUG, "Bucket progress for " + rightCheck +
785                            " will be lost due to merging; potential for duplicates in result-set");
786                }
787                buckets.remove(bucketToKeyWrapper(rightCheck));
788                --pendingBucketCount;
789            }
790        } else {
791            BucketId leftSanityCheck = new BucketId(usedBits, bucket.getId() & ~(1L << (usedBits - 1)));
792            BucketEntry leftSibling = buckets.get(bucketToKeyWrapper(leftSanityCheck));
793            assert(leftSibling == null) : "bucket merge sanity checking failed";
794        }
795
796        BucketId newMerged = new BucketId(usedBits - 1, bucket.getId());
797        addBucket(newMerged, entry.getProgress(), BucketState.BUCKET_PENDING);
798        // Remove original bucket, leaving only the merged bucket
799        buckets.remove(bucketKey);
800        --pendingBucketCount;
801        assert(pendingBucketCount > 0);
802    }
803
804    protected void setAllBucketsToState(BucketState state) {
805        for (Map.Entry<BucketKeyWrapper, ProgressToken.BucketEntry> entry
806                : buckets.entrySet()) {
807            entry.getValue().setState(state);
808        }
809    }
810
811    protected void clearAllBuckets() {
812        buckets.clear();
813        pendingBucketCount = 0;
814        activeBucketCount = 0;
815    }
