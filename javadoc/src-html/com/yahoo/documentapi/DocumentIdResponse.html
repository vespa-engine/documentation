002package com.yahoo.documentapi;
003
004import com.yahoo.document.DocumentId;
005
006/**
007 * The asynchronous response to a document remove operation.
008 * This is a <i>value object</i>.
009 *
010 * @author <a href="mailto:einarmr@yahoo-inc.com">Einar M R Rosenvinge</a>
011 */
012public class DocumentIdResponse extends Response {
013
014    /** The document id of this response, if any */
015    private DocumentId documentId = null;
016
017    /** Creates a successful response */
018    public DocumentIdResponse(long requestId) {
019        super(requestId);
020    }
021
022    /**
023     * Creates a successful response containing a document id
024     *
025     * @param documentId the DocumentId to encapsulate in the Response
026     */
027    public DocumentIdResponse(long requestId, DocumentId documentId) {
028        super(requestId);
029        this.documentId = documentId;
030    }
031
032    /**
033     * Creates a response containing a textual message
034     *
035     * @param textMessage the message to encapsulate in the Response
036     * @param success     true if the response represents a successful call
037     */
038    public DocumentIdResponse(long requestId, String textMessage, boolean success) {
039        super(requestId, textMessage, success);
040    }
041
042    /**
043     * Creates a response containing a textual message and/or a document id
044     *
045     * @param documentId  the DocumentId to encapsulate in the Response
046     * @param textMessage the message to encapsulate in the Response
047     * @param success     true if the response represents a successful call
048     */
049    public DocumentIdResponse(long requestId, DocumentId documentId, String textMessage, boolean success) {
050        super(requestId, textMessage, success);
051        this.documentId = documentId;
052    }
053
054
055    /**
056     * Returns the document id of this response, or null if there is none
057     *
058     * @return the DocumentId, or null
059     */
060    public DocumentId getDocumentId() { return documentId; }
061
062    public int hashCode() {
063        return super.hashCode() + (documentId == null ? 0 : documentId.hashCode());
064    }
065
066    public boolean equals(Object o) {
067        if (!(o instanceof DocumentIdResponse)) {
068            return false;
069        }
070
071        DocumentIdResponse docResp = (DocumentIdResponse) o;
072
073        return super.equals(docResp) && ((documentId == null && docResp.documentId == null) ||
074                (documentId != null && docResp.documentId != null && documentId.equals(docResp.documentId)));
075    }
076
077    public String toString() {
078        return "DocumentId" + super.toString() + (documentId == null ? "" : " " + documentId);
079    }
080
081}
