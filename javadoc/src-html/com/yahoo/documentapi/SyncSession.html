002package com.yahoo.documentapi;
003
004import com.yahoo.document.Document;
005import com.yahoo.document.DocumentId;
006import com.yahoo.document.DocumentPut;
007import com.yahoo.document.DocumentRemove;
008import com.yahoo.document.DocumentType;
009import com.yahoo.document.DocumentUpdate;
010import com.yahoo.document.TestAndSetCondition;
011import com.yahoo.documentapi.messagebus.protocol.DocumentProtocol;
012
013/**
014 * <p>A session for synchronous access to a document repository. This class
015 * provides simple document access where throughput is not a concern.</p>
016 *
017 * @author <a href="mailto:simon@yahoo-inc.com">Simon Thoresen</a>
018 */
019public interface SyncSession extends Session {
020
021    /**
022     * <p>Puts a document. When this method returns, the document is safely
023     * received. This enables setting condition compared to using Document.</p>
024     *
025     * @param documentPut The DocumentPut operation
026     */
027    void put(DocumentPut documentPut);
028
029    /**
030     * <p>Puts a document. When this method returns, the document is safely
031     * received.</p>
032     *
033     * @param documentPut The DocumentPut operation
034     * @param priority The priority with which to perform this operation.
035     */
036    void put(DocumentPut documentPut, DocumentProtocol.Priority priority);
037
038    /**
039     * <p>Gets a document.</p>
040     *
041     * @param id The id of the document to get.
042     * @return The known document having this id, or null if there is no
043     *         document having this id.
044     * @throws UnsupportedOperationException Thrown if this access does not
045     *                                       support retrieving.
046     */
047    Document get(DocumentId id);
048
049    /**
050     * <p>Gets a document.</p>
051     *
052     * @param id       The id of the document to get.
053     * @param fieldSet A comma-separated list of fields to retrieve
054     * @param priority The priority with which to perform this operation.
055     * @return The known document having this id, or null if there is no
056     *         document having this id.
057     * @throws UnsupportedOperationException Thrown if this access does not
058     *                                       support retrieving.
059     */
060    Document get(DocumentId id, String fieldSet, DocumentProtocol.Priority priority);
061
062    /**
063     * <p>Removes a document if it is present and condition is fulfilled.</p>
064     * @param documentRemove document to delete
065     * @return true If the document with this id was removed, false otherwise.
066     */
067    boolean remove(DocumentRemove documentRemove);
068
069    /**
070     * <p>Removes a document if it is present.</p>
071     *
072     * @param documentRemove Document remove operation
073     * @param priority The priority with which to perform this operation.
074     * @return true If the document with this id was removed, false otherwise.
075     * @throws UnsupportedOperationException Thrown if this access does not
076     *                                       support removal.
077     */
078    boolean remove(DocumentRemove documentRemove, DocumentProtocol.Priority priority);
079
080    /**
081     * <p>Updates a document.</p>
082     *
083     * @param update The updates to perform.
084     * @return True, if the document was found and updated.
085     * @throws UnsupportedOperationException Thrown if this access does not
086     *                                       support update.
087     */
088    boolean update(DocumentUpdate update);
089
090    /**
091     * <p>Updates a document.</p>
092     *
093     * @param update   The updates to perform.
094     * @param priority The priority with which to perform this operation.
095     * @return True, if the document was found and updated.
096     * @throws UnsupportedOperationException Thrown if this access does not
097     *                                       support update.
098     */
099    boolean update(DocumentUpdate update, DocumentProtocol.Priority priority);
100
101}
