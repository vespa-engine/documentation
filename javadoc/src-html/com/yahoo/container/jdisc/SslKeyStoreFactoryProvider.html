002package com.yahoo.container.jdisc;
003
004import com.yahoo.container.di.componentgraph.Provider;
005import com.yahoo.jdisc.http.ssl.ReaderForPath;
006import com.yahoo.jdisc.http.ssl.SslKeyStore;
007import com.yahoo.jdisc.http.ssl.SslKeyStoreFactory;
008
009/**
010 * An SSL key store provider which provides a factory which throws exception on
011 * invocation - as no SSL key store is currently provided by default.
012 * The purpose of this is to provide a ssl store factory for injection in the case where
013 * no secret store component is provided.
014 *
015 * @author bratseth
016 */
017public class SslKeyStoreFactoryProvider implements Provider<SslKeyStoreFactory> {
018
019    private static final ThrowingSslKeyStoreFactory instance = new ThrowingSslKeyStoreFactory();
020
021    @Override
022    public SslKeyStoreFactory get() { return instance; }
023
024    @Override
025    public void deconstruct() { }
026
027    private static final class ThrowingSslKeyStoreFactory implements SslKeyStoreFactory {
028
029        @Override
030        public SslKeyStore createKeyStore(ReaderForPath certificateFile, ReaderForPath keyFile) {
031            throw new UnsupportedOperationException("A SSL key store factory component is not available");
032        }
033
034        @Override
035        public SslKeyStore createTrustStore(ReaderForPath certificateFile) {
036            throw new UnsupportedOperationException("A SSL key store factory component is not available");
037        }
038
039    }
040
041}
