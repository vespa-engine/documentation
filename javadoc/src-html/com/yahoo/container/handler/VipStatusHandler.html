002package com.yahoo.container.handler;

004import java.io.File;
005import java.io.FileInputStream;
006import java.io.IOException;
007import java.io.InputStream;
008import java.io.OutputStream;
009import java.util.concurrent.Executor;
010import java.util.concurrent.LinkedBlockingQueue;
011import java.util.concurrent.ThreadPoolExecutor;
012import java.util.concurrent.TimeUnit;
013import java.util.logging.Logger;
014
015import com.google.inject.Inject;
016import com.yahoo.container.core.VipStatusConfig;
017import com.yahoo.container.jdisc.HttpRequest;
018import com.yahoo.container.jdisc.HttpResponse;
019import com.yahoo.container.jdisc.ThreadedHttpRequestHandler;
020import com.yahoo.jdisc.Metric;
021import com.yahoo.log.LogLevel;
022import com.yahoo.text.Utf8;
023import com.yahoo.vespa.defaults.Defaults;
024
025/**
026 * Transmit status to VIP from file or memory. Bind this to
027 * "http://{@literal *}/status.html" to serve VIP status requests.
028 *
029 * @author Steinar Knutsen
030 * @author bratseth
031 */
032public final class VipStatusHandler extends ThreadedHttpRequestHandler {
033
034    private static final Logger log = Logger.getLogger(VipStatusHandler.class.getName());
035
036    private static final String NUM_REQUESTS_METRIC = "jdisc.http.requests.status";
037
038    private final boolean accessDisk;
039    private final File statusFile;
040    private final VipStatus vipStatus;
041    private final boolean noSearchBackendsImpliesOutOfService;
042
043    private volatile boolean previouslyInRotation = true;
044
045    // belongs in the response, but that's not a static class
046    static final String OK_MESSAGE = "<title>OK</title>\n";
047    static final byte[] VIP_OK = Utf8.toBytes(OK_MESSAGE);
048
049    class StatusResponse extends HttpResponse {
050
051        static final String COULD_NOT_FIND_STATUS_FILE = "Could not find status file.\n";
052        static final String NO_SEARCH_BACKENDS = "No search backends available, VIP status disabled.";
053        private static final String TEXT_HTML = "text/html";
054        private String contentType = TEXT_HTML;
055        private byte[] data = null;
056        private File file = null;
057
058        private StatusResponse() {
059            super(com.yahoo.jdisc.http.HttpResponse.Status.OK); // status may be overwritten below
060            if (noSearchBackendsImpliesOutOfService && !vipStatus.isInRotation()) {
061                searchContainerOutOfService();
062            } else if (accessDisk) {
063                preSlurpFile();
064            } else {
065                vipRespond();
066            }
067        }
068
069        @Override
070        public void render(OutputStream stream) throws IOException {
071            if (file != null) {
072                readAndWrite(stream);
073            }
074            else if (data != null) {
075                stream.write(data);
076            }
077            else {
078                throw new IllegalStateException(
079                        "Neither file nor hardcoded data. This is a bug, please notify the Vespa team.");
080            }
081            stream.close();
082        }
083
084        private void readAndWrite(OutputStream stream) throws IOException {
085            InputStream input;
086            int lastRead = 0;
087            input = new FileInputStream(file);
088            try {
089                while (lastRead != -1) {
090                    byte[] buffer = new byte[5000];
091                    lastRead = input.read(buffer);
092                    if (lastRead > 0) {
093                        stream.write(buffer, 0, lastRead);
094                    }
095                }
096            } finally {
097                stream.close();
098                input.close();
099            }
100        }
101
102        private void preSlurpFile() {
103            try {
104                if (!statusFile.exists()) {
105                    fileNotFound();
106                    return;
107                }
108                if (!statusFile.canRead()) {
109                    accessDenied();
110                    return;
111                }
112            } catch (SecurityException e) {
113                internalError();
114                return;
115            }
116            this.file = statusFile;
117        }
118
119        private void accessDenied() {
120            contentType = "text/plain";
121            data = Utf8.toBytes("Status file inaccessible.\n");
122            setStatus(com.yahoo.jdisc.http.HttpResponse.Status.NOT_FOUND);
123        }
124
125        private void internalError() {
126            contentType = "text/plain";
127            data = Utf8.toBytes("Internal error while fetching status file.\n");
128            setStatus(com.yahoo.jdisc.http.HttpResponse.Status.NOT_FOUND);
129        }
130
131        private void fileNotFound() {
132            contentType = "text/plain";
133            data = Utf8.toBytes(COULD_NOT_FIND_STATUS_FILE);
134            setStatus(com.yahoo.jdisc.http.HttpResponse.Status.NOT_FOUND);
135        }
136
137        private void vipRespond() {
138            data = VIP_OK;
139        }
140
141        /**
142         * Behaves like a VIP status response file has been deleted.
143         */
144        private void searchContainerOutOfService() {
145            contentType = "text/plain";
146            data = Utf8.toBytes(NO_SEARCH_BACKENDS);
147            setStatus(com.yahoo.jdisc.http.HttpResponse.Status.NOT_FOUND);
148        }
149
150        @Override
151        public String getContentType() {
152            return contentType;
153        }
154
155        @Override
156        public String getCharacterEncoding() {
157            return null;
158        }
159    }
160
161    /**
162     * Create this with a dedicated thread pool to avoid returning an error to VIPs when the regular thread pool is 
163     * out of capacity. This is the default behavior.
164     */
165    @Inject
166    public VipStatusHandler(VipStatusConfig vipConfig, Metric metric, VipStatus vipStatus) {
167        // One thread should be enough for status handling - otherwise something else is completely wrong,
168        // in which case this will eventually start returning a 503 (due to work rejection) as the bounded
169        // queue will fill up
170        this(new ThreadPoolExecutor(1, 1, 0, TimeUnit.SECONDS, new LinkedBlockingQueue<>(100)),
171             vipConfig, metric, vipStatus);
172    }
173
174    public VipStatusHandler(Executor executor, VipStatusConfig vipConfig, Metric metric) {
175        this(executor, vipConfig, metric, null);
176    }
177
178    public VipStatusHandler(Executor executor, VipStatusConfig vipConfig, Metric metric, VipStatus vipStatus) {
179        super(executor, metric);
180        this.accessDisk = vipConfig.accessdisk();
181        this.statusFile = new File(Defaults.getDefaults().underVespaHome(vipConfig.statusfile()));
182        this.noSearchBackendsImpliesOutOfService = vipConfig.noSearchBackendsImpliesOutOfService();
183        this.vipStatus = vipStatus;
184    }
185
186    @Override
187    public HttpResponse handle(HttpRequest request) {
188        if (metric != null)
189            metric.add(NUM_REQUESTS_METRIC, 1, null);
190        if (noSearchBackendsImpliesOutOfService) {
191            updateAndLogRotationState();
192        }
193        return new StatusResponse();
194    }
195
196    private void updateAndLogRotationState() {
197        final boolean currentlyInRotation = vipStatus.isInRotation();
198        final boolean previousRotationAnswer = previouslyInRotation;
199        previouslyInRotation = currentlyInRotation;
200
201        if (previousRotationAnswer != currentlyInRotation) {
202            if (currentlyInRotation) {
203                log.log(LogLevel.INFO, "Putting container back into rotation by serving status.html again.");
204            } else {
205                log.log(LogLevel.WARNING, "Removing container from rotation by no longer serving status.html.");
206            }
207        }
208    }
