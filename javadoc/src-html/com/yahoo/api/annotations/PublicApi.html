010/**
011 * For annotating packages that should be part of the public API.
012 *
013 * Must be placed in a file called package-info.java in the
014 * package that is to be public.
015 * @author tonytv
016 * @author gjoranv
017 * @since 5.1.5
018 */
019@Retention(RetentionPolicy.RUNTIME)
020@Target(ElementType.PACKAGE)
021@Documented
